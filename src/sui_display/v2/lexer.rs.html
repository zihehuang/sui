7/// Lexer for Display V2 format strings. Format strings are a mix of text and expressions.
8/// Expressions are enclosed in braces and may contain multiple alternates, separated by pipes, and
9/// each containing nested field, vector, or dynamic field accesses.
10#[derive(Debug)]
11pub(crate) struct Lexer<'s> {
12    /// Remaining input to be tokenized.
13    src: &'s str,
14
15    /// The number of bytes (not characters) tokenized so far.
16    off: usize,
17
18    /// Nesting of curly braces. At level 0, the lexer is in text mode. At all other levels, it is
19    /// in expression mode.
20    level: usize,
21}
22
23/// A lexeme is a slice of the source string marked with a token. The `bool` field indicates
24/// whether the lexeme was preceded by whitespace or not.
25#[derive(Copy, Clone, Debug, PartialEq, Eq)]
26pub(crate) struct Lexeme<'s>(pub bool, pub Token, pub usize, pub &'s str);
27
28/// Like [Lexeme] but owns the slice of source string. Useful for capturing context in an error
29/// message.
30#[derive(Debug)]
31pub(crate) struct OwnedLexeme(pub bool, pub Token, pub usize, pub String);
32
33#[derive(Copy, Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
34pub(crate) enum Token {
35    /// '->'
36    Arrow,
37    /// '=>'
38    AArrow,
39    /// '@'
40    At,
41    /// ':'
42    Colon,
43    /// '::'
44    CColon,
45    /// ','
46    Comma,
47    /// '.'
48    Dot,
49    /// An identifier
50    Ident,
51    /// '<'
52    LAngle,
53    /// '{'
54    LBrace,
55    /// '['
56    LBracket,
57    /// '{{'
58    LLBrace,
59    /// '('
60    LParen,
61    /// A decimal number, optionally separated by underscores.
62    NumDec,
63    /// A hexadecimal number, prefixed with '0x' (not included in the span), optionally separated
64    /// by underscores.
65    NumHex,
66    /// '|'
67    Pipe,
68    /// '#'
69    Pound,
70    /// '>'
71    RAngle,
72    /// '}'
73    RBrace,
74    /// ']'
75    RBracket,
76    /// ')'
77    RParen,
78    /// '}}'
79    RRBrace,
80    /// Strings are surrounded by single quotes. Quotes and backslashes inside strings are escaped
81    /// with backslashes.
82    String,
83    /// A strand of text.
84    Text,
85
86    /// An unexpected byte in the input string.
87    Unexpected,
88}

90impl<'s> Lexer<'s> {
91    pub(crate) fn new(src: &'s str) -> Self {
92        Self {
93            src,
94            off: 0,
95            level: 0,
96        }
97    }
98
99    /// Assuming the lexer is in text mode, return the next text token.
100    fn next_text_token(&mut self) -> Option<Lexeme<'s>> {
101        let bytes = self.src.as_bytes();
102
103        use Token as T;
104        Some(match bytes.first()? {
105            b'{' if bytes.get(1) == Some(&b'{') => {
106                self.advance(1);
107                self.take(false, T::LLBrace, 1)
108            }
109
110            b'{' => {
111                self.level += 1;
112                self.take(false, T::LBrace, 1)
113            }
114
115            b'}' if bytes.get(1) == Some(&b'}') => {
116                self.advance(1);
117                self.take(false, T::RRBrace, 1)
118            }
119
120            // This is not a valid token within text, but is recognised so that the parser can
121            // produce a better error message. `level` is not decremenetd because we should already
122            // been in text mode, meaning the level is already 0, and a decrement would underflow
123            // it.
124            b'}' => self.take(false, T::RBrace, 1),
125
126            _ => self.take_until(false, T::Text, |b| b"{}".contains(&b)),
127        })
128    }
129
130    /// Assuming the lexer is in expression mode, return the next expression token.
131    fn next_expr_token(&mut self) -> Option<Lexeme<'s>> {
132        let ws = self.take_whitespace();
133        let bytes = self.src.as_bytes();
134
135        use Token as T;
136        Some(match bytes.first()? {
137            b'-' if bytes.get(1) == Some(&b'>') => self.take(ws, T::Arrow, 2),
138
139            b'=' if bytes.get(1) == Some(&b'>') => self.take(ws, T::AArrow, 2),
140
141            b'@' => self.take(ws, T::At, 1),
142
143            b':' if bytes.get(1) == Some(&b':') => self.take(ws, T::CColon, 2),
144
145            b':' => self.take(ws, T::Colon, 1),
146
147            b',' => self.take(ws, T::Comma, 1),
148
149            b'.' => self.take(ws, T::Dot, 1),
150
151            b'0' if bytes.get(1) == Some(&b'x')
152                && bytes.get(2).is_some_and(|b| is_valid_hex_byte(*b)) =>
153            {
154                self.advance(2);
155                self.take_until(ws, T::NumHex, |c| !is_valid_hex_byte(c))
156            }
157
158            b'0'..=b'9' => self.take_until(ws, T::NumDec, |c| !is_valid_decimal_byte(c)),
159
160            b'a'..=b'z' | b'A'..=b'Z' | b'_' => {
161                self.take_until(ws, T::Ident, |c| !is_valid_identifier_byte(c))
162            }
163
164            b'<' => self.take(ws, T::LAngle, 1),
165
166            b'{' => {
167                self.level += 1;
168                self.take(ws, T::LBrace, 1)
169            }
170
171            b'[' => self.take(ws, T::LBracket, 1),
172
173            b'(' => self.take(ws, T::LParen, 1),
174
175            b'|' => self.take(ws, T::Pipe, 1),
176
177            b'#' => self.take(ws, T::Pound, 1),
178
179            b'>' => self.take(ws, T::RAngle, 1),
180
181            b'}' => {
182                self.level -= 1;
183                self.take(ws, T::RBrace, 1)
184            }
185
186            b']' => self.take(ws, T::RBracket, 1),
187
188            b')' => self.take(ws, T::RParen, 1),
189
190            b'\'' => {
191                // Set the escaped indicator to true initially so we don't interpret the starting
192                // quote as an ending quote.
193                let mut escaped = true;
194                for (i, b) in self.src.bytes().enumerate() {
195                    if escaped {
196                        escaped = false;
197                    } else if b == b'\\' {
198                        escaped = true;
199                    } else if b == b'\'' {
200                        self.advance(1);
201                        let content = self.take(ws, T::String, i - 1);
202                        self.advance(1);
203                        return Some(content);
204                    }
205                }
206
207                // Reached the end of the byte stream and didn't find a closing quote -- treat the
208                // partial string as an unexpected token.
209                self.take(ws, T::Unexpected, self.src.len())
210            }
211
212            // If the next byte cannot be recognized, extract the next (potentially variable
213            // length) character, and indicate that it is an unexpected token.
214            _ => {
215                let next_boundary = (1..=self.src.len())
216                    .find(|&i| self.src.is_char_boundary(i))
217                    .unwrap_or(self.src.len());
218                self.take(ws, T::Unexpected, next_boundary)
219            }
220        })
221    }
222
223    /// Eat ASCII whitespace from the beginning of `self.src`. Returns `true` if and only if any
224    /// whitespace was consumed.
225    fn take_whitespace(&mut self) -> bool {
226        let Lexeme(_, _, _, slice) =
227            self.take_until(false, Token::Unexpected, |b| !b.is_ascii_whitespace());
228        !slice.is_empty()
229    }
230
231    /// Take a prefix of bytes from `self.src` until a byte satisfying pattern `p` is found, and
232    /// return it as a lexeme of type `t`. If no such byte is found, take the entire remainder of
233    /// the source string.
234    fn take_until(&mut self, ws: bool, t: Token, p: impl FnMut(u8) -> bool) -> Lexeme<'s> {
235        let n = self.src.bytes().position(p).unwrap_or(self.src.len());
236        self.take(ws, t, n)
237    }
238
239    /// Take `n` bytes from the beginning of `self.src` and return them as a lexeme of type `t`.
240    ///
241    /// ## Safety
242    ///
243    /// This function assumes that there are at least `n` bytes left in `self.src`, and will panic
244    /// if that is not the case.
245    fn take(&mut self, ws: bool, t: Token, n: usize) -> Lexeme<'s> {
246        let start = self.off;
247        let slice = &self.src[..n];
248        self.advance(n);
249
250        Lexeme(ws, t, start, slice)
251    }
252
253    /// Move the cursor forward by `n` bytes.
254    ///
255    /// ## Safety
256    ///
257    /// This function assumes that `n` is less than or equal to the length of `self.src`, and will
258    /// panic if that is not the case.
259    fn advance(&mut self, n: usize) {
260        self.src = &self.src[n..];
261        self.off += n;
262    }
263}
264
265impl Lexeme<'_> {
266    /// Return the lexeme as an owned lexeme, with the slice of source string copied.
267    pub(crate) fn detach(&self) -> OwnedLexeme {
268        OwnedLexeme(self.0, self.1, self.2, self.3.to_owned())
269    }
270}

272impl<'s> Iterator for Lexer<'s> {
273    type Item = Lexeme<'s>;
274
275    fn next(&mut self) -> Option<Self::Item> {
276        if self.level == 0 {
277            self.next_text_token()
278        } else {
279            self.next_expr_token()
280        }
281    }
282}

284impl fmt::Display for OwnedLexeme {
285    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
286        use OwnedLexeme as L;
287        use Token as T;
288
289        if self.0 {
290            write!(f, "whitespace followed by ")?;
291        }
292
293        match self {
294            L(_, T::Arrow, _, _) => write!(f, "'->'"),
295            L(_, T::AArrow, _, _) => write!(f, "'=>'"),
296            L(_, T::At, _, _) => write!(f, "'@'"),
297            L(_, T::Colon, _, _) => write!(f, "':'"),
298            L(_, T::CColon, _, _) => write!(f, "'::'"),
299            L(_, T::Comma, _, _) => write!(f, "','"),
300            L(_, T::Dot, _, _) => write!(f, "'.'"),
301            L(_, T::Ident, _, s) => write!(f, "identifier {s:?}"),
302            L(_, T::LAngle, _, _) => write!(f, "'<'"),
303            L(_, T::LBrace, _, _) => write!(f, "'{{'"),
304            L(_, T::LBracket, _, _) => write!(f, "'['"),
305            L(_, T::LLBrace, _, _) => write!(f, "'{{{{'"),
306            L(_, T::LParen, _, _) => write!(f, "'('"),
307            L(_, T::NumDec, _, s) => write!(f, "decimal number {s:?}"),
308            L(_, T::NumHex, _, s) => write!(f, "hexadecimal number {s:?}"),
309            L(_, T::Pipe, _, _) => write!(f, "'|'"),
310            L(_, T::Pound, _, _) => write!(f, "'#'"),
311            L(_, T::RAngle, _, _) => write!(f, "'>'"),
312            L(_, T::RBrace, _, _) => write!(f, "'}}'"),
313            L(_, T::RBracket, _, _) => write!(f, "']'"),
314            L(_, T::RParen, _, _) => write!(f, "')'"),
315            L(_, T::RRBrace, _, _) => write!(f, "'}}}}'"),
316            L(_, T::String, _, s) => write!(f, "string {s:?}"),
317            L(_, T::Text, _, s) => write!(f, "text {s:?}"),
318            L(_, T::Unexpected, _, s) => {
319                write!(f, "\"")?;
320                for b in s.bytes() {
321                    match b {
322                        b'"' => write!(f, "\\\"")?,
323                        b'\\' => write!(f, "\\\\")?,
324                        b'\n' => write!(f, "\\n")?,
325                        b'\t' => write!(f, "\\t")?,
326                        b'\r' => write!(f, "\\r")?,
327                        b if b.is_ascii_graphic() || b == b' ' => write!(f, "{}", b as char)?,
328                        b => write!(f, "\\x{:02X}", b)?,
329                    }
330                }
331                write!(f, "\"")
332            }
333        }?;
334
335        write!(f, " at offset {}", self.2)
336    }
337}

339impl fmt::Display for Token {
340    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
341        use Token as T;
342        match self {
343            T::Arrow => write!(f, "'->'"),
344            T::AArrow => write!(f, "'=>'"),
345            T::At => write!(f, "'@'"),
346            T::Colon => write!(f, "':'"),
347            T::CColon => write!(f, "'::'"),
348            T::Comma => write!(f, "','"),
349            T::Dot => write!(f, "'.'"),
350            T::Ident => write!(f, "an identifier"),
351            T::LAngle => write!(f, "'<'"),
352            T::LBrace => write!(f, "'{{'"),
353            T::LBracket => write!(f, "'['"),
354            T::LLBrace => write!(f, "'{{{{'"),
355            T::LParen => write!(f, "'('"),
356            T::NumDec => write!(f, "a decimal number"),
357            T::NumHex => write!(f, "a hexadecimal number"),
358            T::Pipe => write!(f, "'|'"),
359            T::Pound => write!(f, "'#'"),
360            T::RAngle => write!(f, "'>'"),
361            T::RBrace => write!(f, "'}}'"),
362            T::RBracket => write!(f, "']'"),
363            T::RParen => write!(f, "')'"),
364            T::RRBrace => write!(f, "'}}}}'"),
365            T::String => write!(f, "a string"),
366            T::Text => write!(f, "text"),
367            T::Unexpected => write!(f, "unexpected input"),
368        }
369    }
370}

372fn is_valid_identifier_byte(b: u8) -> bool {
373    matches!(b, b'_' | b'a'..=b'z' | b'A'..=b'Z' | b'0'..=b'9')
374}
375
376fn is_valid_hex_byte(b: u8) -> bool {
377    matches!(b, b'_' | b'a'..=b'f' | b'A'..=b'F' | b'0'..=b'9')
378}
379
380fn is_valid_decimal_byte(b: u8) -> bool {
381    matches!(b, b'_' | b'0'..=b'9')
382}

390    fn lexemes(src: &str) -> String {
391        Lexer::new(src)
392            .map(|L(ws, t, o, s)| {
393                // Handle potentially invalid UTF-8 by working at byte level
394                let safe_s: String = s
395                    .bytes()
396                    .map(|b| match b {
397                        b'"' => "\\\"".to_string(),
398                        b'\\' => "\\\\".to_string(),
399                        b'\n' => "\\n".to_string(),
400                        b'\t' => "\\t".to_string(),
401                        b'\r' => "\\r".to_string(),
402                        b if b.is_ascii_graphic() || b == b' ' => (b as char).to_string(),
403                        b => format!("\\x{:02X}", b),
404                    })
405                    .collect();
406                format!("L({ws:?}, {t:?}, {o:?}, \"{}\")", safe_s)
407            })
408            .collect::<Vec<_>>()
409            .join("\n")
410    }

887    /// Test the UTF-8 boundary fallback when string ends with incomplete UTF-8.
888    /// This exercises the .unwrap_or(self.src.len()) fallback in the boundary detection.
889    #[test]
890    fn test_incomplete_utf8_boundary_fallback() {
891        // Create input with incomplete UTF-8: '{' + first byte of multi-byte sequence
892        let mut input = vec![b'{'];
893        input.push(0xC3); // First byte of multi-byte UTF-8 sequence (missing continuation)
894        let input_str = unsafe { std::str::from_utf8_unchecked(&input) };
895
896        // The boundary detection should fall back to src.len() when no boundary is found
897        assert_snapshot!(lexemes(input_str), @r###"
898        L(false, LBrace, 0, "{")
899        L(false, Unexpected, 1, "\xC3")
900        "###);
