121#[derive(Clone)]
122pub struct NetworkAuthorityClient {
123    client: SuiResult<ValidatorClient<Channel>>,
124}
125
126impl NetworkAuthorityClient {
127    pub async fn connect(
128        address: &Multiaddr,
129        tls_target: NetworkPublicKey,
130    ) -> anyhow::Result<Self> {
131        let tls_config = sui_tls::create_rustls_client_config(
132            tls_target,
133            sui_tls::SUI_VALIDATOR_SERVER_NAME.to_string(),
134            None,
135        );
136        let channel = mysten_network::client::connect(address, tls_config)
137            .await
138            .map_err(|err| anyhow!(err.to_string()))?;
139        Ok(Self::new(channel))
140    }
141
142    pub fn connect_lazy(address: &Multiaddr, tls_target: NetworkPublicKey) -> Self {
143        let tls_config = sui_tls::create_rustls_client_config(
144            tls_target,
145            sui_tls::SUI_VALIDATOR_SERVER_NAME.to_string(),
146            None,
147        );
148        let client: SuiResult<_> = mysten_network::client::connect_lazy(address, tls_config)
149            .map(ValidatorClient::new)
150            .map_err(|err| err.to_string().into());
151        Self { client }
152    }
153
154    pub fn new(channel: Channel) -> Self {
155        Self {
156            client: Ok(ValidatorClient::new(channel)),
157        }
158    }
159
160    fn new_lazy(client: SuiResult<Channel>) -> Self {
161        Self {
162            client: client.map(ValidatorClient::new),
163        }
164    }
165
166    pub(crate) fn client(&self) -> SuiResult<ValidatorClient<Channel>> {
167        self.client.clone()
168    }
169}

174    async fn submit_transaction(
175        &self,
176        request: SubmitTxRequest,
177        client_addr: Option<SocketAddr>,
178    ) -> Result<SubmitTxResponse, SuiError> {
179        let mut request = request.into_raw()?.into_request();
180        insert_metadata(&mut request, client_addr);
181
182        self.client()?

348pub fn make_network_authority_clients_with_network_config(
349    committee: &CommitteeWithNetworkMetadata,
350    network_config: &Config,
351) -> BTreeMap<AuthorityName, NetworkAuthorityClient> {
352    let mut authority_clients = BTreeMap::new();
353    for (name, (_state, network_metadata)) in committee.validators() {
354        let address = network_metadata
355            .network_address
356            .clone()
357            .rewrite_udp_to_tcp()
358            .rewrite_http_to_https();
359        let tls_config = network_metadata
360            .network_public_key
361            .as_ref()
362            .map(|key| {
363                sui_tls::create_rustls_client_config(
364                    key.clone(),
365                    sui_tls::SUI_VALIDATOR_SERVER_NAME.to_string(),
366                    None,
367                )
368            })
369            .ok_or(SuiError::from("network public key is not available"));
370        let maybe_channel = tls_config
371            .and_then(|tls_config| {
372                network_config
373                    .connect_lazy(&address, tls_config)
374                    .map_err(|e| e.to_string().into())
375            })
376            .tap_err(|e| {
377                tracing::error!(
378                    address = %address,
379                    name = %name,
380                    "unable to create authority client: {e}"
381                )
382            });
383        let client = NetworkAuthorityClient::new_lazy(maybe_channel);
384        authority_clients.insert(*name, client);
385    }
386    authority_clients
387}

389pub fn make_authority_clients_with_timeout_config(
390    committee: &CommitteeWithNetworkMetadata,
391    connect_timeout: Duration,
392    request_timeout: Duration,
393) -> BTreeMap<AuthorityName, NetworkAuthorityClient> {
394    let mut network_config = mysten_network::config::Config::new();
395    network_config.connect_timeout = Some(connect_timeout);
396    network_config.request_timeout = Some(request_timeout);
397    network_config.http2_keepalive_interval = Some(connect_timeout);
398    network_config.http2_keepalive_timeout = Some(connect_timeout);
399    make_network_authority_clients_with_network_config(committee, &network_config)
400}

402fn insert_metadata<T>(request: &mut tonic::Request<T>, client_addr: Option<SocketAddr>) {
403    if let Some(client_addr) = client_addr {
404        let mut metadata = tonic::metadata::MetadataMap::new();
405        metadata.insert("x-forwarded-for", client_addr.to_string().parse().unwrap());
406        metadata
407            .iter()
408            .for_each(|key_and_value| match key_and_value {
409                KeyAndValueRef::Ascii(key, value) => {
410                    request.metadata_mut().insert(key, value.clone());
411                }
412                KeyAndValueRef::Binary(key, value) => {
413                    request.metadata_mut().insert_bin(key, value.clone());
414                }
415            });
416    }
