111impl<A> TransactionOrchestrator<A>
112where
113    A: AuthorityAPI + Send + Sync + 'static + Clone,
114    OnsiteReconfigObserver: ReconfigObserver<A>,
115{
116    pub fn new(
117        validators: Arc<AuthorityAggregator<A>>,
118        validator_state: Arc<AuthorityState>,
119        parent_path: &Path,
120        prometheus_registry: &Registry,
121        reconfig_observer: OnsiteReconfigObserver,
122        node_config: &NodeConfig,
123    ) -> Self {
124        let metrics = Arc::new(QuorumDriverMetrics::new(prometheus_registry));
125        let notifier = Arc::new(NotifyRead::new());
126        let reconfig_observer = Arc::new(reconfig_observer);
127        let quorum_driver_handler = Arc::new(
128            QuorumDriverHandlerBuilder::new(validators.clone(), metrics.clone())
129                .with_notifier(notifier.clone())
130                .with_reconfig_observer(reconfig_observer.clone())
131                .start(),
132        );
133
134        let effects_receiver = quorum_driver_handler.subscribe_to_effects();
135        let metrics = Arc::new(TransactionOrchestratorMetrics::new(prometheus_registry));
136        let pending_tx_log = Arc::new(WritePathPendingTransactionLog::new(
137            parent_path.join("fullnode_pending_transactions"),
138        ));
139        let pending_tx_log_clone = pending_tx_log.clone();
140        let _local_executor_handle = {
141            spawn_monitored_task!(async move {
142                Self::loop_pending_transaction_log(effects_receiver, pending_tx_log_clone).await;
143            })
144        };
145        Self::schedule_txes_in_log(pending_tx_log.clone(), quorum_driver_handler.clone());
146
147        let epoch_store = validator_state.load_epoch_store_one_call_per_task();
148        let td_percentage = if !epoch_store.protocol_config().mysticeti_fastpath() {
149            0
150        } else {
151            choose_transaction_driver_percentage(Some(epoch_store.get_chain_identifier()))
152        };
153
154        let transaction_driver = if td_percentage > 0 {
155            let td_metrics = Arc::new(TransactionDriverMetrics::new(prometheus_registry));
156            let client_metrics = Arc::new(
157                crate::validator_client_monitor::ValidatorClientMetrics::new(prometheus_registry),
158            );
159            Some(TransactionDriver::new(
160                validators.clone(),
161                reconfig_observer.clone(),
162                td_metrics,
163                Some(node_config),
164                client_metrics,
165            ))
166        } else {
167            None
168        };
169
170        let td_allowed_submission_list = node_config
171            .transaction_driver_config
172            .as_ref()
173            .map(|config| config.allowed_submission_validators.clone())
174            .unwrap_or_default();
175
176        Self {
177            quorum_driver_handler,
178            validator_state,
179            _local_executor_handle,
180            pending_tx_log,
181            notifier,
182            metrics,
183            transaction_driver,
184            td_percentage,
185            td_allowed_submission_list,
186        }
187    }

199    pub async fn execute_transaction_block(
200        &self,
201        request: ExecuteTransactionRequestV3,
202        request_type: ExecuteTransactionRequestType,
203        client_addr: Option<SocketAddr>,
204    ) -> Result<(ExecuteTransactionResponseV3, IsTransactionExecutedLocally), QuorumDriverError>
205    {
206        let timer = Instant::now();
207        let tx_type = if request.transaction.is_consensus_tx() {
208            TxType::SharedObject
209        } else {
210            TxType::SingleWriter
211        };
212        let tx_digest = *request.transaction.digest();
213
214        let (response, mut executed_locally) = self
215            .execute_transaction_with_effects_waiting(request, client_addr)
216            .await?;
217
218        if !executed_locally {
219            executed_locally = if matches!(
220                request_type,
221                ExecuteTransactionRequestType::WaitForLocalExecution
222            ) {
223                let executed_locally = Self::wait_for_finalized_tx_executed_locally_with_timeout(
224                    &self.validator_state,
225                    tx_digest,
226                    tx_type,
227                    &self.metrics,
228                )
229                .await
230                .is_ok();
231                add_server_timing("local_execution done");
232                executed_locally
233            } else {
234                false
235            };
236        }
237
238        let QuorumTransactionResponse {
239            effects,
240            events,
241            input_objects,
242            output_objects,
243            auxiliary_data,
244        } = response;
245
246        let response = ExecuteTransactionResponseV3 {
247            effects,
248            events,
249            input_objects,
250            output_objects,
251            auxiliary_data,
252        };
253
254        self.metrics
255            .request_latency
256            .with_label_values(&[
257                tx_type.as_str(),
258                "execute_transaction_block",
259                executed_locally.to_string().as_str(),
260            ])
261            .observe(timer.elapsed().as_secs_f64());
262
263        Ok((response, executed_locally))
264    }

307    /// Shared implementation for executing transactions with parallel local effects waiting
308    async fn execute_transaction_with_effects_waiting(
309        &self,
310        request: ExecuteTransactionRequestV3,
311        client_addr: Option<SocketAddr>,
312    ) -> Result<(QuorumTransactionResponse, IsTransactionExecutedLocally), QuorumDriverError> {
313        let epoch_store = self.validator_state.load_epoch_store_one_call_per_task();
314        let verified_transaction = epoch_store
315            .verify_transaction(request.transaction.clone())
316            .map_err(QuorumDriverError::InvalidUserSignature)?;
317        let tx_digest = *verified_transaction.digest();
318
319        // Add transaction to WAL log.
320        let is_new_transaction = self
321            .pending_tx_log
322            .write_pending_transaction_maybe(&verified_transaction)
323            .await
324            .map_err(|e| {
325                warn!("QuorumDriverInternalError: {e:?}");
326                QuorumDriverError::QuorumDriverInternalError(e)
327            })?;
328        if is_new_transaction {
329            debug!("Added transaction to WAL log for TransactionDriver");
330        } else {
331            debug!("Transaction already in pending_tx_log");
332        }
333
334        let include_events = request.include_events;
335        let include_input_objects = request.include_input_objects;
336        let include_output_objects = request.include_output_objects;
337        let include_auxiliary_data = request.include_auxiliary_data;
338
339        // Track whether TD is being used for this transaction
340        let using_td = Arc::new(AtomicBool::new(false));
341
342        let finality_timeout = std::env::var("WAIT_FOR_FINALITY_TIMEOUT_SECS")
343            .ok()
344            .and_then(|v| v.parse().ok())
345            .map(Duration::from_secs)
346            .unwrap_or(WAIT_FOR_FINALITY_TIMEOUT);
347
348        let num_submissions = if !is_new_transaction {
349            // No need to submit when the transaction is already being processed.
350            0
351        } else if cfg!(msim) || in_antithesis() {
352            // Allow duplicated submissions in tests.
353            let r = rand::thread_rng().gen_range(1..=100);
354            let n = if r <= 10 {
355                3
356            } else if r <= 30 {
357                2
358            } else {
359                1
360            };
361            if n > 1 {
362                debug!("Making {n} execution calls");
363            }
364            n
365        } else {
366            1
367        };
368
369        // Wait for one of the execution futures to succeed, or all of them to fail.
370        let mut execution_futures = FuturesUnordered::new();
371        for i in 0..num_submissions {
372            // Generate jitter values outside the async block
373            let should_delay = i > 0 && rand::thread_rng().gen_bool(0.8);
374            let delay_ms = if should_delay {
375                rand::thread_rng().gen_range(100..=500)
376            } else {
377                0
378            };
379
380            let epoch_store = epoch_store.clone();
381            let request = request.clone();
382            let verified_transaction = verified_transaction.clone();
383            let using_td = using_td.clone();
384
385            let future = async move {
386                if delay_ms > 0 {
387                    // Add jitters to duplicated submissions.
388                    sleep(Duration::from_millis(delay_ms)).await;
389                }
390                self.execute_transaction_impl(
391                    &epoch_store,
392                    request,
393                    verified_transaction,
394                    client_addr,
395                    Some(finality_timeout),
396                    using_td,
397                )
398                .await
399            }
400            .boxed();
401            execution_futures.push(future);
402        }
403
404        // Track the last execution error.
405        let mut last_execution_error: Option<QuorumDriverError> = None;
406
407        // Wait for execution result outside of this call to become available.
408        let digests = [tx_digest];
409        let mut local_effects_future = epoch_store
410            .within_alive_epoch(
411                self.validator_state
412                    .get_transaction_cache_reader()
413                    .notify_read_executed_effects(
414                    "TransactionOrchestrator::notify_read_execute_transaction_with_effects_waiting",
415                    &digests,
416                ),
417            )
418            .boxed();
419
420        // Wait for execution timeout.
421        let mut timeout_future = tokio::time::sleep(finality_timeout).boxed();
422
423        let result = loop {
424            tokio::select! {
425                biased;
426
427                // Local effects might be available
428                local_effects_result = &mut local_effects_future => {
429                    match local_effects_result {
430                        Ok(effects) => {
431                            debug!(
432                                "Effects became available while execution was running"
433                            );
434                            if let Some(effects) = effects.into_iter().next() {
435                                self.metrics.concurrent_execution.inc();
436                                let epoch = effects.executed_epoch();
437                                let events = if include_events {
438                                    if effects.events_digest().is_some() {
439                                        Some(self.validator_state.get_transaction_events(effects.transaction_digest())
440                                            .map_err(QuorumDriverError::QuorumDriverInternalError)?)
441                                    } else {
442                                        None
443                                    }
444                                } else {
445                                    None
446                                };
447                                let input_objects = include_input_objects
448                                    .then(|| self.validator_state.get_transaction_input_objects(&effects))
449                                    .transpose()
450                                    .map_err(QuorumDriverError::QuorumDriverInternalError)?;
451                                let output_objects = include_output_objects
452                                    .then(|| self.validator_state.get_transaction_output_objects(&effects))
453                                    .transpose()
454                                    .map_err(QuorumDriverError::QuorumDriverInternalError)?;
455                                let response = QuorumTransactionResponse {
456                                    effects: FinalizedEffects {
457                                        effects,
458                                        finality_info: EffectsFinalityInfo::QuorumExecuted(epoch),
459                                    },
460                                    events,
461                                    input_objects,
462                                    output_objects,
463                                    auxiliary_data: None,
464                                };
465                                break Ok((response, true));
466                            }
467                        }
468                        Err(_) => {
469                            warn!("Epoch terminated before effects were available");
470                        }
471                    };
472
473                    // Prevent this branch from being selected again
474                    local_effects_future = futures::future::pending().boxed();
475                }
476
477                // This branch is disabled if execution_futures is empty.
478                Some(result) = execution_futures.next() => {
479                    match result {
480                        Ok(resp) => {
481                            // First success gets returned.
482                            debug!("Execution succeeded, returning response");
483                            let QuorumTransactionResponse {
484                                effects,
485                                events,
486                                input_objects,
487                                output_objects,
488                                auxiliary_data,
489                            } = resp;
490                            // Filter fields based on request flags.
491                            let resp = QuorumTransactionResponse {
492                                effects,
493                                events: if include_events { events } else { None },
494                                input_objects: if include_input_objects { input_objects } else { None },
495                                output_objects: if include_output_objects { output_objects } else { None },
496                                auxiliary_data: if include_auxiliary_data { auxiliary_data } else { None },
497                            };
498                            break Ok((resp, false));
499                        }
500                        Err(QuorumDriverError::PendingExecutionInTransactionOrchestrator) => {
501                            debug!(
502                                "Transaction is already being processed"
503                            );
504                            // Avoid overriding errors with transaction already being processed.
505                            if last_execution_error.is_none() {
506                                last_execution_error = Some(QuorumDriverError::PendingExecutionInTransactionOrchestrator);
507                            }
508                        }
509                        Err(e) => {
510                            debug!(?e, "Execution attempt failed, wait for other attempts");
511                            last_execution_error = Some(e);
512                        }
513                    };
514
515                    // Last error must have been recorded.
516                    if execution_futures.is_empty() {
517                        break Err(last_execution_error.unwrap());
518                    }
519                }
520
521                // A timeout has occurred while waiting for finality
522                _ = &mut timeout_future => {
523                    debug!("Timeout waiting for transaction finality.");
524                    self.metrics.wait_for_finality_timeout.inc();
525
526                    // Clean up transaction from WAL log only for TD submissions
527                    // For QD submissions, the cleanup happens in loop_pending_transaction_log
528                    if using_td.load(Ordering::Acquire) {
529                        debug!("Cleaning up TD transaction from WAL due to timeout");
530                        if let Err(err) = self.pending_tx_log.finish_transaction(&tx_digest) {
531                            warn!(
532                                "Failed to finish TD transaction in pending transaction log: {err}"
533                            );
534                        }
535                    }
536
537                    break Err(QuorumDriverError::TimeoutBeforeFinality);
538                }
539            }
540        };
541
542        // Clean up transaction from WAL log
543        if let Err(err) = self.pending_tx_log.finish_transaction(&tx_digest) {
544            warn!("Failed to finish transaction in pending transaction log: {err}");
545        }
546
547        result
548    }

550    #[instrument(level = "error", skip_all)]
551    async fn execute_transaction_impl(
552        &self,
553        epoch_store: &Arc<AuthorityPerEpochStore>,
554        request: ExecuteTransactionRequestV3,
555        verified_transaction: VerifiedTransaction,
556        client_addr: Option<SocketAddr>,
557        finality_timeout: Option<Duration>,
558        using_td: Arc<AtomicBool>,
559    ) -> Result<QuorumTransactionResponse, QuorumDriverError> {
560        let tx_digest = *verified_transaction.digest();
561        debug!("TO Received transaction execution request.");
562
563        let timer = Instant::now();
564        let tx_type = if verified_transaction.is_consensus_tx() {
565            TxType::SharedObject
566        } else {
567            TxType::SingleWriter
568        };
569
570        let (_in_flight_metrics_guards, good_response_metrics) =
571            self.update_metrics(&request.transaction);
572
573        // TODO: refactor all the gauge and timer metrics with `monitored_scope`
574        let wait_for_finality_gauge = self.metrics.wait_for_finality_in_flight.clone();
575        wait_for_finality_gauge.inc();
576        let _wait_for_finality_gauge = scopeguard::guard(wait_for_finality_gauge, |in_flight| {
577            in_flight.dec();
578        });
579
580        // Select TransactionDriver or QuorumDriver for submission.
581        let (response, driver_type) = if self.transaction_driver.is_some()
582            && self.should_use_transaction_driver(epoch_store, tx_digest)
583        {
584            // Mark that we're using TD before submitting.
585            using_td.store(true, Ordering::Release);
586
587            (
588                self.submit_with_transaction_driver(
589                    self.transaction_driver.as_ref().unwrap(),
590                    &request,
591                    client_addr,
592                    &verified_transaction,
593                    good_response_metrics,
594                    finality_timeout,
595                )
596                .await?,
597                "transaction_driver",
598            )
599        } else {
600            // Submit transaction through QuorumDriver.
601            using_td.store(false, Ordering::Release);
602
603            let resp = self
604                .submit_with_quorum_driver(
605                    epoch_store.clone(),
606                    verified_transaction.clone(),
607                    request,
608                    client_addr,
609                )
610                .await
611                .map_err(|e| {
612                    warn!("QuorumDriverInternalError: {e:?}");
613                    QuorumDriverError::QuorumDriverInternalError(e)
614                })?
615                .await
616                .map_err(|e| {
617                    warn!("QuorumDriverInternalError: {e:?}");
618                    QuorumDriverError::QuorumDriverInternalError(e)
619                })??;
620
621            (
622                QuorumTransactionResponse {
623                    effects: FinalizedEffects::new_from_effects_cert(resp.effects_cert.into()),
624                    events: resp.events,
625                    input_objects: resp.input_objects,
626                    output_objects: resp.output_objects,
627                    auxiliary_data: resp.auxiliary_data,
628                },
629                "quorum_driver",
630            )
631        };
632
633        add_server_timing("wait_for_finality done");
634
635        self.metrics.wait_for_finality_finished.inc();
636
637        let elapsed = timer.elapsed().as_secs_f64();
638        self.metrics
639            .settlement_finality_latency
640            .with_label_values(&[tx_type.as_str(), driver_type])
641            .observe(elapsed);
642        good_response_metrics.inc();
643
644        Ok(response)
645    }

647    #[instrument(level = "error", skip_all, err(level = "info"))]
648    async fn submit_with_transaction_driver(
649        &self,
650        td: &Arc<TransactionDriver<A>>,
651        request: &ExecuteTransactionRequestV3,
652        client_addr: Option<SocketAddr>,
653        verified_transaction: &VerifiedTransaction,
654        good_response_metrics: &GenericCounter<AtomicU64>,
655        timeout_duration: Option<Duration>,
656    ) -> Result<QuorumTransactionResponse, QuorumDriverError> {
657        let tx_digest = *verified_transaction.digest();
658        debug!("Using TransactionDriver for transaction {:?}", tx_digest);
659
660        let td_response = td
661            .drive_transaction(
662                SubmitTxRequest::new_transaction(request.transaction.clone()),
663                SubmitTransactionOptions {
664                    forwarded_client_addr: client_addr,
665                    allowed_validators: self.td_allowed_submission_list.clone(),
666                },
667                timeout_duration,
668            )
669            .await
670            .map_err(|e| match e {
671                TransactionDriverError::TimeoutWithLastRetriableError {
672                    last_error,
673                    attempts,
674                    timeout,
675                } => QuorumDriverError::TimeoutBeforeFinalityWithErrors {
676                    last_error: last_error.map(|e| e.to_string()).unwrap_or_default(),
677                    attempts,
678                    timeout,
679                },
680                other => QuorumDriverError::TransactionFailed {
681                    category: other.categorize(),
682                    details: other.to_string(),
683                },
684            });
685
686        match td_response {
687            Err(e) => {
688                warn!("TransactionDriver error: {e:?}");
689                Err(e)
690            }
691            Ok(quorum_transaction_response) => {
692                good_response_metrics.inc();
693                Ok(quorum_transaction_response)
694            }
695        }
696    }

698    /// Submits the transaction to Quorum Driver for execution.
699    /// Returns an awaitable Future.
700    #[instrument(name = "tx_orchestrator_submit", level = "trace", skip_all)]
701    async fn submit_with_quorum_driver(
702        &self,
703        epoch_store: Arc<AuthorityPerEpochStore>,
704        transaction: VerifiedTransaction,
705        request: ExecuteTransactionRequestV3,
706        client_addr: Option<SocketAddr>,
707    ) -> SuiResult<impl Future<Output = SuiResult<QuorumDriverResult>> + '_> {
708        let tx_digest = *transaction.digest();
709
710        let ticket = self.notifier.register_one(&tx_digest);
711        self.quorum_driver()
712            .submit_transaction_no_ticket(request.clone(), client_addr)
713            .await?;
714
715        // It's possible that the transaction effects is already stored in DB at this point.
716        // So we also subscribe to that. If we hear from `effects_await` first, it means
717        // the ticket misses the previous notification, and we want to ask quorum driver
718        // to form a certificate for us again, to serve this request.
719        let cache_reader = self.validator_state.get_transaction_cache_reader().clone();
720        let qd = self.clone_quorum_driver();
721        Ok(async move {
722            let digests = [tx_digest];
723            let effects_await =
724                epoch_store.within_alive_epoch(cache_reader.notify_read_executed_effects(
725                    "TransactionOrchestrator::notify_read_submit_with_qd",
726                    &digests,
727                ));
728            // let-and-return necessary to satisfy borrow checker.
729            #[allow(clippy::let_and_return)]
730            let res = match select(ticket, effects_await.boxed()).await {
731                Either::Left((quorum_driver_response, _)) => Ok(quorum_driver_response),
732                Either::Right((_, unfinished_quorum_driver_task)) => {
733                    debug!("Effects are available in DB, use quorum driver to get a certificate");
734                    qd.submit_transaction_no_ticket(request, client_addr)
735                        .await?;
736                    Ok(unfinished_quorum_driver_task.await)
737                }
738            };
739            res
740        })
741    }

743    #[instrument(
744        name = "tx_orchestrator_wait_for_finalized_tx_executed_locally_with_timeout",
745        level = "debug",
746        skip_all,
747        err(level = "info")
748    )]
749    async fn wait_for_finalized_tx_executed_locally_with_timeout(
750        validator_state: &Arc<AuthorityState>,
751        tx_digest: TransactionDigest,
752        tx_type: TxType,
753        metrics: &TransactionOrchestratorMetrics,
754    ) -> SuiResult {
755        metrics.local_execution_in_flight.inc();
756        let _metrics_guard =
757            scopeguard::guard(metrics.local_execution_in_flight.clone(), |in_flight| {
758                in_flight.dec();
759            });
760
761        let _latency_guard = metrics
762            .local_execution_latency
763            .with_label_values(&[tx_type.as_str()])
764            .start_timer();
765        debug!("Waiting for finalized tx to be executed locally.");
766        match timeout(
767            LOCAL_EXECUTION_TIMEOUT,
768            validator_state
769                .get_transaction_cache_reader()
770                .notify_read_executed_effects_digests(
771                    "TransactionOrchestrator::notify_read_wait_for_local_execution",
772                    &[tx_digest],
773                ),
774        )
775        .instrument(error_span!(
776            "transaction_orchestrator::local_execution",
777            ?tx_digest
778        ))
779        .await
780        {
781            Err(_elapsed) => {
782                debug!(
783                    "Waiting for finalized tx to be executed locally timed out within {:?}.",
784                    LOCAL_EXECUTION_TIMEOUT
785                );
786                metrics.local_execution_timeout.inc();
787                Err(SuiErrorKind::TimeoutError.into())
788            }
789            Ok(_) => {
790                metrics.local_execution_success.inc();
791                Ok(())
792            }
793        }
794    }

796    fn should_use_transaction_driver(
797        &self,
798        epoch_store: &Arc<AuthorityPerEpochStore>,
799        tx_digest: TransactionDigest,
800    ) -> bool {
801        const MAX_PERCENTAGE: u8 = 100;
802        let unknown_network = epoch_store.get_chain() == Chain::Unknown;
803        let v = if unknown_network {
804            rand::thread_rng().gen_range(1..=MAX_PERCENTAGE)
805        } else {
806            let v = u32::from_le_bytes(tx_digest.inner()[..4].try_into().unwrap());
807            (v % (MAX_PERCENTAGE as u32) + 1) as u8
808        };
809        debug!(
810            "Choosing whether to use transaction driver: {} vs {}",
811            v, self.td_percentage
812        );
813        v <= self.td_percentage
814    }

816    // TODO: Potentially cleanup this function and pending transaction log.
817    async fn loop_pending_transaction_log(
818        mut effects_receiver: Receiver<QuorumDriverEffectsQueueResult>,
819        pending_transaction_log: Arc<WritePathPendingTransactionLog>,
820    ) {
821        loop {
822            match effects_receiver.recv().await {
823                Ok(Ok((transaction, ..))) => {
824                    let tx_digest = transaction.digest();
825                    if let Err(err) = pending_transaction_log.finish_transaction(tx_digest) {
826                        error!(
827                            ?tx_digest,
828                            "Failed to finish transaction in pending transaction log: {err}"
829                        );
830                    }
831                }
832                Ok(Err((tx_digest, _err))) => {
833                    if let Err(err) = pending_transaction_log.finish_transaction(&tx_digest) {
834                        error!(
835                            ?tx_digest,
836                            "Failed to finish transaction in pending transaction log: {err}"
837                        );
838                    }
839                }
840                Err(RecvError::Closed) => {
841                    error!("Sender of effects subscriber queue has been dropped!");
842                    return;
843                }
844                Err(RecvError::Lagged(skipped_count)) => {
845                    warn!("Skipped {skipped_count} transasctions in effects subscriber queue.");
846                }
847            }
848        }
849    }

889    fn schedule_txes_in_log(
890        pending_tx_log: Arc<WritePathPendingTransactionLog>,
891        quorum_driver: Arc<QuorumDriverHandler<A>>,
892    ) {
893        spawn_logged_monitored_task!(async move {
894            if std::env::var("SKIP_LOADING_FROM_PENDING_TX_LOG").is_ok() {
895                info!("Skipping loading pending transactions from pending_tx_log.");
896                return;
897            }
898            let pending_txes = pending_tx_log
899                .load_all_pending_transactions()
900                .expect("failed to load all pending transactions");
901            info!(
902                "Recovering {} pending transactions from pending_tx_log.",
903                pending_txes.len()
904            );
905            for (i, tx) in pending_txes.into_iter().enumerate() {
906                // TODO: ideally pending_tx_log would not contain VerifiedTransaction, but that
907                // requires a migration.
908                let tx = tx.into_inner();
909                let tx_digest = *tx.digest();
910                // It's not impossible we fail to enqueue a task but that's not the end of world.
911                // TODO(william) correctly extract client_addr from logs
912                if let Err(err) = quorum_driver
913                    .submit_transaction_no_ticket(
914                        ExecuteTransactionRequestV3 {
915                            transaction: tx,
916                            include_events: true,
917                            include_input_objects: false,
918                            include_output_objects: false,
919                            include_auxiliary_data: false,
920                        },
921                        None,
922                    )
923                    .await
924                {
925                    warn!(
926                        ?tx_digest,
927                        "Failed to enqueue transaction from pending_tx_log, err: {err:?}"
928                    );
929                } else {
930                    debug!("Enqueued transaction from pending_tx_log");
931                    if (i + 1) % 1000 == 0 {
932                        info!("Enqueued {} transactions from pending_tx_log.", i + 1);
933                    }
934                }
935            }
936            // Transactions will be cleaned up in loop_execute_finalized_tx_locally() after they
937            // produce effects.
938        });
939    }
