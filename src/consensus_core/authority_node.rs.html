45/// ConsensusAuthority is used by Sui to manage the lifetime of AuthorityNode.
46/// It hides the details of the implementation from the caller, MysticetiManager.
47#[allow(private_interfaces)]
48pub enum ConsensusAuthority {
49    WithAnemo(AuthorityNode<AnemoManager>),
50    WithTonic(AuthorityNode<TonicManager>),
51}
52
53impl ConsensusAuthority {
54    pub async fn start(
55        network_type: ConsensusNetwork,
56        epoch_start_timestamp_ms: u64,
57        own_index: AuthorityIndex,
58        committee: Committee,
59        parameters: Parameters,
60        protocol_config: ProtocolConfig,
61        protocol_keypair: ProtocolKeyPair,
62        network_keypair: NetworkKeyPair,
63        clock: Arc<Clock>,
64        transaction_verifier: Arc<dyn TransactionVerifier>,
65        commit_consumer: CommitConsumerArgs,
66        registry: Registry,
67        // A counter that keeps track of how many times the authority node has been booted while the binary
68        // or the component that is calling the `ConsensusAuthority` has been running. It's mostly useful to
69        // make decisions on whether amnesia recovery should run or not. When `boot_counter` is 0, then `ConsensusAuthority`
70        // will initiate the process of amnesia recovery if that's enabled in the parameters.
71        boot_counter: u64,
72    ) -> Self {
73        match network_type {
74            ConsensusNetwork::Anemo => {
75                let authority = AuthorityNode::start(
76                    epoch_start_timestamp_ms,
77                    own_index,
78                    committee,
79                    parameters,
80                    protocol_config,
81                    protocol_keypair,
82                    network_keypair,
83                    clock,
84                    transaction_verifier,
85                    commit_consumer,
86                    registry,
87                    boot_counter,
88                )
89                .await;
90                Self::WithAnemo(authority)
91            }
92            ConsensusNetwork::Tonic => {
93                let authority = AuthorityNode::start(
94                    epoch_start_timestamp_ms,
95                    own_index,
96                    committee,
97                    parameters,
98                    protocol_config,
99                    protocol_keypair,
100                    network_keypair,
101                    clock,
102                    transaction_verifier,
103                    commit_consumer,
104                    registry,
105                    boot_counter,
106                )
107                .await;
108                Self::WithTonic(authority)
109            }
110        }
111    }
112
113    pub async fn stop(self) {
114        match self {
115            Self::WithAnemo(authority) => authority.stop().await,
116            Self::WithTonic(authority) => authority.stop().await,
117        }
118    }
119
120    pub fn transaction_client(&self) -> Arc<TransactionClient> {
121        match self {
122            Self::WithAnemo(authority) => authority.transaction_client(),
123            Self::WithTonic(authority) => authority.transaction_client(),
124        }
125    }
126
127    #[cfg(test)]
128    fn context(&self) -> &Arc<Context> {
129        match self {
130            Self::WithAnemo(authority) => &authority.context,
131            Self::WithTonic(authority) => &authority.context,
132        }
133    }
134
135    #[allow(unused)]
136    fn sync_last_known_own_block_enabled(&self) -> bool {
137        match self {
138            Self::WithAnemo(authority) => authority.sync_last_known_own_block,
139            Self::WithTonic(authority) => authority.sync_last_known_own_block,
140        }
141    }

170    pub(crate) async fn start(
171        epoch_start_timestamp_ms: u64,
172        own_index: AuthorityIndex,
173        committee: Committee,
174        parameters: Parameters,
175        protocol_config: ProtocolConfig,
176        // To avoid accidentally leaking the private key, the protocol key pair should only be
177        // kept in Core.
178        protocol_keypair: ProtocolKeyPair,
179        network_keypair: NetworkKeyPair,
180        clock: Arc<Clock>,
181        transaction_verifier: Arc<dyn TransactionVerifier>,
182        commit_consumer: CommitConsumerArgs,
183        registry: Registry,
184        boot_counter: u64,
185    ) -> Self {
186        assert!(
187            committee.is_valid_index(own_index),
188            "Invalid own index {}",
189            own_index
190        );
191        let own_hostname = committee.authority(own_index).hostname.clone();
192        info!(
193            "Starting consensus authority {} {}, {:?}, epoch start timestamp {}, boot counter {}, replaying after commit index {}, consumer last processed commit index {}",
194            own_index,
195            own_hostname,
196            protocol_config.version,
197            epoch_start_timestamp_ms,
198            boot_counter,
199            commit_consumer.replay_after_commit_index,
200            commit_consumer.consumer_last_processed_commit_index
201        );
202        info!(
203            "Consensus authorities: {}",
204            committee
205                .authorities()
206                .map(|(i, a)| format!("{}: {}", i, a.hostname))
207                .join(", ")
208        );
209        info!("Consensus parameters: {:?}", parameters);
210        info!("Consensus committee: {:?}", committee);
211        let context = Arc::new(Context::new(
212            epoch_start_timestamp_ms,
213            own_index,
214            committee,
215            parameters,
216            protocol_config,
217            initialise_metrics(registry),
218            clock,
219        ));
220        let start_time = Instant::now();
221
222        context
223            .metrics
224            .node_metrics
225            .authority_index
226            .with_label_values(&[&own_hostname])
227            .set(context.own_index.value() as i64);
228        context
229            .metrics
230            .node_metrics
231            .protocol_version
232            .set(context.protocol_config.version.as_u64() as i64);
233
234        let (tx_client, tx_receiver) = TransactionClient::new(context.clone());
235        let tx_consumer = TransactionConsumer::new(tx_receiver, context.clone());
236
237        let (core_signals, signals_receivers) = CoreSignals::new(context.clone());
238
239        let mut network_manager = N::new(context.clone(), network_keypair);
240        let network_client = network_manager.client();
241
242        // REQUIRED: Broadcaster must be created before Core, to start listening on the
243        // broadcast channel in order to not miss blocks and cause test failures.
244        let broadcaster = if N::Client::SUPPORT_STREAMING {
245            None
246        } else {
247            Some(Broadcaster::new(
248                context.clone(),
249                network_client.clone(),
250                &signals_receivers,
251            ))
252        };
253
254        let store_path = context.parameters.db_path.as_path().to_str().unwrap();
255        let store = Arc::new(RocksDBStore::new(store_path));
256        let dag_state = Arc::new(RwLock::new(DagState::new(context.clone(), store.clone())));
257
258        let block_verifier = Arc::new(SignedBlockVerifier::new(
259            context.clone(),
260            transaction_verifier,
261        ));
262
263        let transaction_certifier = TransactionCertifier::new(
264            context.clone(),
265            dag_state.clone(),
266            commit_consumer.block_sender.clone(),
267        );
268        // TODO(fastpath): recover incrementally by chunk inside CommitObserver::new()
269        transaction_certifier.recover(
270            block_verifier.as_ref(),
271            commit_consumer.replay_after_commit_index,
272        );
273
274        let mut proposed_block_handler = ProposedBlockHandler::new(
275            context.clone(),
276            signals_receivers.block_broadcast_receiver(),
277            transaction_certifier.clone(),
278        );
279
280        let proposed_block_handler =
281            spawn_logged_monitored_task!(proposed_block_handler.run(), "proposed_block_handler");
282
283        let sync_last_known_own_block = boot_counter == 0
284            && dag_state.read().highest_accepted_round() == 0
285            && !context
286                .parameters
287                .sync_last_known_own_block_timeout
288                .is_zero();
289        info!("Sync last known own block: {sync_last_known_own_block}");
290
291        let block_manager = BlockManager::new(context.clone(), dag_state.clone());
292
293        let leader_schedule = Arc::new(LeaderSchedule::from_store(
294            context.clone(),
295            dag_state.clone(),
296        ));
297
298        let commit_consumer_monitor = commit_consumer.monitor();
299        let commit_observer = CommitObserver::new(
300            context.clone(),
301            commit_consumer,
302            dag_state.clone(),
303            transaction_certifier.clone(),
304            leader_schedule.clone(),
305        )
306        .await;
307
308        let round_tracker = Arc::new(RwLock::new(PeerRoundTracker::new(context.clone())));
309
310        let core = Core::new(
311            context.clone(),
312            leader_schedule,
313            tx_consumer,
314            transaction_certifier.clone(),
315            block_manager,
316            // For streaming RPC, Core will be notified when consumer is available.
317            // For non-streaming RPC, there is no way to know so default to true.
318            // When there is only one (this) authority, assume subscriber exists.
319            !N::Client::SUPPORT_STREAMING || context.committee.size() == 1,
320            commit_observer,
321            core_signals,
322            protocol_keypair,
323            dag_state.clone(),
324            sync_last_known_own_block,
325            round_tracker.clone(),
326        );
327
328        let (core_dispatcher, core_thread_handle) =
329            ChannelCoreThreadDispatcher::start(context.clone(), &dag_state, core);
330        let core_dispatcher = Arc::new(core_dispatcher);
331        let leader_timeout_handle =
332            LeaderTimeoutTask::start(core_dispatcher.clone(), &signals_receivers, context.clone());
333
334        let commit_vote_monitor = Arc::new(CommitVoteMonitor::new(context.clone()));
335
336        let synchronizer = Synchronizer::start(
337            network_client.clone(),
338            context.clone(),
339            core_dispatcher.clone(),
340            commit_vote_monitor.clone(),
341            block_verifier.clone(),
342            transaction_certifier.clone(),
343            dag_state.clone(),
344            sync_last_known_own_block,
345        );
346
347        let commit_syncer_handle = CommitSyncer::new(
348            context.clone(),
349            core_dispatcher.clone(),
350            commit_vote_monitor.clone(),
351            commit_consumer_monitor.clone(),
352            block_verifier.clone(),
353            transaction_certifier.clone(),
354            network_client.clone(),
355            dag_state.clone(),
356        )
357        .start();
358
359        let round_prober_handle = RoundProber::new(
360            context.clone(),
361            core_dispatcher.clone(),
362            round_tracker.clone(),
363            dag_state.clone(),
364            network_client.clone(),
365        )
366        .start();
367
368        let network_service = Arc::new(AuthorityService::new(
369            context.clone(),
370            block_verifier,
371            commit_vote_monitor,
372            round_tracker.clone(),
373            synchronizer.clone(),
374            core_dispatcher,
375            signals_receivers.block_broadcast_receiver(),
376            transaction_certifier,
377            dag_state.clone(),
378            store,
379        ));
380
381        let subscriber = if N::Client::SUPPORT_STREAMING {
382            let s = Subscriber::new(
383                context.clone(),
384                network_client,
385                network_service.clone(),
386                dag_state,
387            );
388            for (peer, _) in context.committee.authorities() {
389                if peer != context.own_index {
390                    s.subscribe(peer);
391                }
392            }
393            Some(s)
394        } else {
395            None
396        };
397
398        network_manager.install_service(network_service).await;
399
400        info!(
401            "Consensus authority started, took {:?}",
402            start_time.elapsed()
403        );
404
405        Self {
406            context,
407            start_time,
408            transaction_client: Arc::new(tx_client),
409            synchronizer,
410            commit_syncer_handle,
411            round_prober_handle,
412            proposed_block_handler,
413            leader_timeout_handle,
414            core_thread_handle,
415            broadcaster,
416            subscriber,
417            network_manager,
418            sync_last_known_own_block,
419        }
420    }
421
422    pub(crate) async fn stop(mut self) {
423        info!(
424            "Stopping authority. Total run time: {:?}",
425            self.start_time.elapsed()
426        );
427
428        // First shutdown components calling into Core.
429        if let Err(e) = self.synchronizer.stop().await {
430            if e.is_panic() {
431                std::panic::resume_unwind(e.into_panic());
432            }
433            warn!(
434                "Failed to stop synchronizer when shutting down consensus: {:?}",
435                e
436            );
437        };
438        self.commit_syncer_handle.stop().await;
439        self.round_prober_handle.stop().await;
440        self.proposed_block_handler.abort();
441        self.leader_timeout_handle.stop().await;
442        // Shutdown Core to stop block productions and broadcast.
443        // When using streaming, all subscribers to broadcasted blocks stop after this.
444        self.core_thread_handle.stop().await;
445        if let Some(mut broadcaster) = self.broadcaster.take() {
446            broadcaster.stop();
447        }
448        // Stop outgoing long lived streams before stopping network server.
449        if let Some(subscriber) = self.subscriber.take() {
450            subscriber.stop();
451        }
452        self.network_manager.stop().await;
453
454        self.context
455            .metrics
456            .node_metrics
457            .uptime
458            .observe(self.start_time.elapsed().as_secs_f64());
459    }

461    pub(crate) fn transaction_client(&self) -> Arc<TransactionClient> {
462        self.transaction_client.clone()
463    }

707                        );
708                    }
709                }
710                assert_eq!(committed_subdag.reputation_scores_desc, vec![]);
711                if expected_transactions.is_empty() {
712                    break;
713                }
714            }
715        }
716
717        // Stop authority 0.
718        let index = committee.to_authority_index(0).unwrap();
719        authorities.remove(index.value()).stop().await;
720        sleep(Duration::from_secs(10)).await;
721
722        // Restart authority 0 and let it run.
723        let (authority, commit_receiver, _block_receiver) = make_authority(
724            index,
725            &temp_dirs[index.value()],
726            committee.clone(),
727            keypairs.clone(),
728            network_type,
729            boot_counters[index],
730            protocol_config.clone(),
731        )
732        .await;
733        boot_counters[index] += 1;
734        output_receivers[index] = commit_receiver;
735        authorities.insert(index.value(), authority);
736        sleep(Duration::from_secs(10)).await;
737
738        // Stop all authorities and exit.
739        for authority in authorities {
740            authority.stop().await;
741        }

745    #[tokio::test(flavor = "current_thread")]
746    async fn test_amnesia_recovery_success(#[values(5, 10)] gc_depth: u32) {
747        telemetry_subscribers::init_for_testing();
748        let db_registry = Registry::new();
749        DBMetrics::init(RegistryService::new(db_registry));
750
751        const NUM_OF_AUTHORITIES: usize = 4;
752        let (committee, keypairs) = local_committee_and_keys(0, [1; NUM_OF_AUTHORITIES].to_vec());
753        let mut commit_receivers = vec![];
754        let mut block_receivers = vec![];
755        let mut authorities = BTreeMap::new();
756        let mut temp_dirs = BTreeMap::new();
757        let mut boot_counters = [0; NUM_OF_AUTHORITIES];
758
759        let mut protocol_config = ProtocolConfig::get_for_max_version_UNSAFE();
760        protocol_config.set_consensus_gc_depth_for_testing(gc_depth);
761
762        for (index, _authority_info) in committee.authorities() {
763            let dir = TempDir::new().unwrap();
764            let (authority, commit_receiver, block_receiver) = make_authority(
765                index,
766                &dir,
767                committee.clone(),
768                keypairs.clone(),
769                ConsensusNetwork::Tonic,
770                boot_counters[index],
771                protocol_config.clone(),
772            )
773            .await;
774            assert!(
775                authority.sync_last_known_own_block_enabled(),
776                "Expected syncing of last known own block to be enabled as all authorities are of empty db and boot for first time."
777            );
778            boot_counters[index] += 1;
779            commit_receivers.push(commit_receiver);
780            block_receivers.push(block_receiver);
781            authorities.insert(index, authority);
782            temp_dirs.insert(index, dir);
783        }
784
785        // Now we take the receiver of authority 1 and we wait until we see at least one block committed from this authority
786        // We wait until we see at least one committed block authored from this authority. That way we'll be 100% sure that
787        // at least one block has been proposed and successfully received by a quorum of nodes.
788        let index_1 = committee.to_authority_index(1).unwrap();
789        'outer: while let Some(result) =
790            timeout(Duration::from_secs(10), commit_receivers[index_1].recv())
791                .await
792                .expect("Timed out while waiting for at least one committed block from authority 1")
793        {
794            for block in result.blocks {
795                if block.round() > GENESIS_ROUND && block.author() == index_1 {
796                    break 'outer;
797                }
798            }
799        }
800
801        // Stop authority 1 & 2.
802        // * Authority 1 will be used to wipe out their DB and practically "force" the amnesia recovery.
803        // * Authority 2 is stopped in order to simulate less than f+1 availability which will
804        // make authority 1 retry during amnesia recovery until it has finally managed to successfully get back f+1 responses.
805        // once authority 2 is up and running again.
806        authorities.remove(&index_1).unwrap().stop().await;
807        let index_2 = committee.to_authority_index(2).unwrap();
808        authorities.remove(&index_2).unwrap().stop().await;
809        sleep(Duration::from_secs(5)).await;
810
811        // Authority 1: create a new directory to simulate amnesia. The node will start having participated previously
812        // to consensus but now will attempt to synchronize the last own block and recover from there. It won't be able
813        // to do that successfully as authority 2 is still down.
814        let dir = TempDir::new().unwrap();
815        // We do reset the boot counter for this one to simulate a "binary" restart
816        boot_counters[index_1] = 0;
817        let (authority, mut commit_receiver, _block_receiver) = make_authority(
818            index_1,
819            &dir,
820            committee.clone(),
821            keypairs.clone(),
822            ConsensusNetwork::Tonic,
823            boot_counters[index_1],
824            protocol_config.clone(),
825        )
826        .await;
827        assert!(
828            authority.sync_last_known_own_block_enabled(),
829            "Authority should have the sync of last own block enabled"
830        );
831        boot_counters[index_1] += 1;
832        authorities.insert(index_1, authority);
833        temp_dirs.insert(index_1, dir);
834        sleep(Duration::from_secs(5)).await;
835
836        // Now spin up authority 2 using its earlier directly - so no amnesia recovery should be forced here.
837        // Authority 1 should be able to recover from amnesia successfully.
838        let (authority, _commit_receiver, _block_receiver) = make_authority(
839            index_2,
840            &temp_dirs[&index_2],
841            committee.clone(),
842            keypairs,
843            ConsensusNetwork::Tonic,
844            boot_counters[index_2],
845            protocol_config.clone(),
846        )
847        .await;
848        assert!(
849            !authority.sync_last_known_own_block_enabled(),
850            "Authority should not have attempted to sync the last own block"
851        );
852        boot_counters[index_2] += 1;
853        authorities.insert(index_2, authority);
854        sleep(Duration::from_secs(5)).await;
855
856        // We wait until we see at least one committed block authored from this authority
857        'outer: while let Some(result) = commit_receiver.recv().await {
858            for block in result.blocks {
859                if block.round() > GENESIS_ROUND && block.author() == index_1 {
860                    break 'outer;
861                }
862            }
863        }
864
865        // Stop all authorities and exit.
866        for (_, authority) in authorities {
867            authority.stop().await;
868        }
869    }
