28/// For transaction T committed at leader round R, when a new leader at round >= R + INDIRECT_REJECT_DEPTH
29/// commits and T is still not finalized, T is rejected.
30/// NOTE: 3 round is the minimum depth possible for indirect finalization and rejection.
31pub(crate) const INDIRECT_REJECT_DEPTH: Round = 3;
32
33/// Handle to CommitFinalizer, for sending CommittedSubDag.
34pub(crate) struct CommitFinalizerHandle {
35    sender: UnboundedSender<CommittedSubDag>,
36}
37
38impl CommitFinalizerHandle {
39    // Sends a CommittedSubDag to CommitFinalizer, which will finalize it before sending it to execution.
40    pub(crate) fn send(&self, commit: CommittedSubDag) -> ConsensusResult<()> {
41        self.sender.send(commit).map_err(|e| {
42            tracing::warn!("Failed to send to commit finalizer, probably due to shutdown: {e:?}");
43            ConsensusError::Shutdown
44        })
45    }
46}

48/// CommitFinalizer accepts a continuous stream of CommittedSubDag and outputs
49/// them when they are finalized.
50/// In finalized commits, every transaction is either finalized or rejected.
51/// It runs in a separate thread, to reduce the load on the core thread.
52///
53/// Life of a finalized commit:
54///
55/// For efficiency, finalization happens first for transactions without reject votes (common case).
56/// The pending undecided transactions with reject votes are individually finalized or rejected.
57/// When there is no more pending transactions, the commit is finalized.
58///
59/// This is correct because regardless if a commit leader was directly or indirectly committed,
60/// every committed block can be considered finalized, because at least one leader certificate of the commit
61/// will be committed, which can also serve as a certificate for the block and its transactions.
62///
63/// From the earliest buffered commit, pending blocks are checked to see if they are now finalized.
64/// New finalized blocks are removed from the pending blocks, and its transactions are moved to the
65/// finalized, rejected or pending state. If the commit now has no pending blocks or transactions,
66/// the commit is finalized and popped from the buffer. The next earliest commit is then processed
67/// similarly, until either the buffer becomes empty or a commit with pending blocks or transactions
68/// is encountered.
69pub struct CommitFinalizer {
70    context: Arc<Context>,
71    dag_state: Arc<RwLock<DagState>>,
72    transaction_certifier: TransactionCertifier,
73    commit_sender: UnboundedSender<CommittedSubDag>,
74
75    // Last commit index processed by CommitFinalizer.
76    last_processed_commit: Option<CommitIndex>,
77    // Commits pending finalization.
78    pending_commits: VecDeque<CommitState>,
79    // Blocks in the pending commits.
80    blocks: Arc<RwLock<BTreeMap<BlockRef, RwLock<BlockState>>>>,
81}

101    pub(crate) fn start(
102        context: Arc<Context>,
103        dag_state: Arc<RwLock<DagState>>,
104        transaction_certifier: TransactionCertifier,
105        commit_sender: UnboundedSender<CommittedSubDag>,
106    ) -> CommitFinalizerHandle {
107        let processor = Self::new(context, dag_state, transaction_certifier, commit_sender);
108        let (sender, receiver) = unbounded_channel("consensus_commit_finalizer");
109        let _handle =
110            spawn_logged_monitored_task!(processor.run(receiver), "consensus_commit_finalizer");
111        CommitFinalizerHandle { sender }
112    }

114    async fn run(mut self, mut receiver: UnboundedReceiver<CommittedSubDag>) {
115        while let Some(committed_sub_dag) = receiver.recv().await {
116            let finalized_commits = if self.context.protocol_config.mysticeti_fastpath() {
117                self.process_commit(committed_sub_dag).await
118            } else {
119                vec![committed_sub_dag]
120            };
121            if !finalized_commits.is_empty() {
122                let mut dag_state = self.dag_state.write();
123                if self.context.protocol_config.mysticeti_fastpath() {
124                    // Records commits that have been finalized and their rejected transactions.
125                    for commit in &finalized_commits {
126                        dag_state.add_finalized_commit(
127                            commit.commit_ref,
128                            commit.rejected_transactions_by_block.clone(),
129                        );
130                    }
131                }
132                // Commits and committed blocks must be persisted to storage before sending them to Sui
133                // to execute their finalized transactions.
134                // Commit metadata and uncommitted blocks can be persisted more lazily because they are recoverable.
135                // But for simplicity, all unpersisted commits and blocks are flushed to storage.
136                dag_state.flush();
137            }
138            for commit in finalized_commits {
139                if let Err(e) = self.commit_sender.send(commit) {
140                    tracing::warn!(
141                        "Failed to send to commit handler, probably due to shutdown: {e:?}"
142                    );
143                    return;
144                }
145            }
146        }
147    }

149    pub async fn process_commit(
150        &mut self,
151        committed_sub_dag: CommittedSubDag,
152    ) -> Vec<CommittedSubDag> {
153        let _scope = monitored_scope("CommitFinalizer::process_commit");
154
155        if let Some(last_processed_commit) = self.last_processed_commit {
156            assert_eq!(
157                last_processed_commit + 1,
158                committed_sub_dag.commit_ref.index
159            );
160        }
161        self.last_processed_commit = Some(committed_sub_dag.commit_ref.index);
162
163        self.pending_commits
164            .push_back(CommitState::new(committed_sub_dag));
165
166        let mut finalized_commits = vec![];
167
168        // The prerequisite for running direct finalization on a commit is that the commit must
169        // have either a quorum of leader certificates in the local DAG, or a committed leader certificate.
170        //
171        // A leader certificate is a finalization certificate for every block in the commit.
172        // When the prerequisite holds, all blocks in the current commit can be considered finalized.
173        // And any transaction in the current commit that has not observed reject votes will never be rejected.
174        // So these transactions are directly finalized.
175        //
176        // When a commit is direct, there are a quorum of its leader certificates in the local DAG.
177        //
178        // When a commit is indirect, it implies one of its leader certificates is in the committed blocks.
179        // So a leader certificate must exist in the local DAG as well.
180        //
181        // When a commit is received through commit sync and processed as certified commit, the commit might
182        // not have a leader certificate in the local DAG. So a committed transaction might not observe any reject
183        // vote from local DAG, although it will eventually get rejected. To finalize blocks in this commit,
184        // there must be another commit with leader round >= 3 (WAVE_LENGTH) rounds above the commit leader.
185        // From the indirect commit rule, a leader certificate must exist in committed blocks for the earliest commit.
186        for i in 0..self.pending_commits.len() {
187            let commit_state = &self.pending_commits[i];
188            if commit_state.pending_blocks.is_empty() {
189                // The commit has already been processed through direct finalization.
190                continue;
191            }
192            // Direct finalization cannot happen when
193            // -  This commit is remote.
194            // -  And the latest commit is less than 3 (WAVE_LENGTH) rounds above this commit.
195            // In this case, this commit's leader certificate is not guaranteed to be in local DAG.
196            if !commit_state.commit.decided_with_local_blocks {
197                let last_commit_state = self.pending_commits.back().unwrap();
198                if commit_state.commit.leader.round + DEFAULT_WAVE_LENGTH
199                    > last_commit_state.commit.leader.round
200                {
201                    break;
202                }
203            }
204            self.try_direct_finalize_commit(i);
205        }
206        let direct_finalized_commits = self.pop_finalized_commits();
207        self.context
208            .metrics
209            .node_metrics
210            .finalizer_output_commits
211            .with_label_values(&["direct"])
212            .inc_by(direct_finalized_commits.len() as u64);
213        finalized_commits.extend(direct_finalized_commits);
214
215        // Indirect finalization: one or more commits cannot be directly finalized.
216        // So the pending transactions need to be checked for indirect finalization.
217        if !self.pending_commits.is_empty() {
218            // Initialize the state of the last added commit for computing indirect finalization.
219            //
220            // As long as there are remaining commits, even if the last commit has been directly finalized,
221            // its state still needs to be initialized here to help indirectly finalize previous commits.
222            // This is because the last commit may have been directly finalized, but its previous commits
223            // may not have been directly finalized.
224            self.link_blocks_in_last_commit();
225            self.append_origin_descendants_from_last_commit();
226            // Try to indirectly finalize a prefix of the buffered commits.
227            // If only one commit remains, it cannot be indirectly finalized because there is no commit afterwards,
228            // so it is excluded.
229            while self.pending_commits.len() > 1 {
230                // Stop indirect finalization when the earliest commit has not been processed
231                // through direct finalization.
232                if !self.pending_commits[0].pending_blocks.is_empty() {
233                    break;
234                }
235                // Otherwise, try to indirectly finalize the earliest commit.
236                self.try_indirect_finalize_first_commit().await;
237                let indirect_finalized_commits = self.pop_finalized_commits();
238                if indirect_finalized_commits.is_empty() {
239                    // No additional commits can be indirectly finalized.
240                    break;
241                }
242                self.context
243                    .metrics
244                    .node_metrics
245                    .finalizer_output_commits
246                    .with_label_values(&["indirect"])
247                    .inc_by(indirect_finalized_commits.len() as u64);
248                finalized_commits.extend(indirect_finalized_commits);
249            }
250        }
251
252        // GC TransactionCertifier state only with finalized commits, to ensure unfinalized transactions
253        // can access their reject votes from TransactionCertifier.
254        if let Some(last_commit) = finalized_commits.last() {
255            let gc_round = self
256                .dag_state
257                .read()
258                .calculate_gc_round(last_commit.leader.round);
259            self.transaction_certifier.run_gc(gc_round);
260        }
261
262        self.context
263            .metrics
264            .node_metrics
265            .finalizer_buffered_commits
266            .set(self.pending_commits.len() as i64);
267
268        finalized_commits
269    }

271    // Tries directly finalizing transactions in the commit.
272    fn try_direct_finalize_commit(&mut self, index: usize) {
273        let num_commits = self.pending_commits.len();
274        let commit_state = self
275            .pending_commits
276            .get_mut(index)
277            .unwrap_or_else(|| panic!("Commit {} does not exist. len = {}", index, num_commits,));
278        // Direct commit means every transaction in the commit can be considered to have a quorum of post-commit certificates,
279        // unless the transaction has reject votes that do not reach quorum either.
280        assert!(!commit_state.pending_blocks.is_empty());
281
282        let metrics = &self.context.metrics.node_metrics;
283        let pending_blocks = std::mem::take(&mut commit_state.pending_blocks);
284        for (block_ref, num_transactions) in pending_blocks {
285            let reject_votes = self.transaction_certifier.get_reject_votes(&block_ref)
286                .unwrap_or_else(|| panic!("No vote info found for {block_ref}. It is either incorrectly gc'ed or failed to be recovered after crash."));
287            metrics
288                .finalizer_transaction_status
289                .with_label_values(&["direct_finalize"])
290                .inc_by((num_transactions - reject_votes.len()) as u64);
291            let hostname = &self.context.committee.authority(block_ref.author).hostname;
292            metrics
293                .finalizer_reject_votes
294                .with_label_values(&[hostname])
295                .inc_by(reject_votes.len() as u64);
296            // If a transaction_index does not exist in reject_votes, the transaction has no reject votes.
297            // So it is finalized and does not need to be added to pending_transactions.
298            for (transaction_index, stake) in reject_votes {
299                // If the transaction has > 0 but < 2f+1 reject votes, it is still pending.
300                // Otherwise, it is rejected.
301                let entry = if stake < self.context.committee.quorum_threshold() {
302                    commit_state
303                        .pending_transactions
304                        .entry(block_ref)
305                        .or_default()
306                } else {
307                    metrics
308                        .finalizer_transaction_status
309                        .with_label_values(&["direct_reject"])
310                        .inc();
311                    commit_state
312                        .rejected_transactions
313                        .entry(block_ref)
314                        .or_default()
315                };
316                entry.insert(transaction_index);
317            }
318        }
319    }

321    // Creates an entry in the blocks map for each block in the commit,
322    // and have its ancestors link to the block.
323    fn link_blocks_in_last_commit(&mut self) {
324        let commit_state = self
325            .pending_commits
326            .back_mut()
327            .unwrap_or_else(|| panic!("No pending commit."));
328
329        // Link blocks in ascending order of round, to ensure ancestor block states are created
330        // before they are linked from.
331        let mut blocks = commit_state.commit.blocks.clone();
332        blocks.sort_by_key(|b| b.round());
333
334        let mut blocks_map = self.blocks.write();
335        for block in blocks {
336            let block_ref = block.reference();
337            // Link ancestors to the block.
338            for ancestor in block.ancestors() {
339                // Ancestor may not exist in the blocks map if it has been finalized or gc'ed.
340                // So skip linking if the ancestor does not exist.
341                if let Some(ancestor_block) = blocks_map.get(ancestor) {
342                    ancestor_block.write().children.insert(block_ref);
343                }
344            }
345            // Initialize the block state.
346            blocks_map
347                .entry(block_ref)
348                .or_insert_with(|| RwLock::new(BlockState::new(block)));
349        }
350    }

352    /// To save bandwidth, blocks do not include explicit accept votes on transactions.
353    /// Reject votes are included only the first time the block containing the voted-on
354    /// transaction is linked in a block. Other first time linked transactions, when
355    /// not rejected, are assumed to be accepted. This vote compression rule must also be
356    /// applied during vote aggregation.
357    ///
358    /// Transactions in a block can only be voted on by its immediate descendants.
359    /// A block is an **immediate descendant** if it can only link directly to the voted-on
360    /// block, without any intermediate blocks from its own authority. Votes from
361    /// non-immediate descendants are ignored.
362    ///
363    /// This rule implies the following optimization is possible: after collecting votes from a block,
364    /// we can skip collecting votes from its **origin descendants** (descendant blocks from the
365    /// same authority), because their votes would be ignored anyway.
366    ///
367    /// This function updates the set of origin descendants for all pending blocks using blocks
368    /// from the last commit.
369    fn append_origin_descendants_from_last_commit(&mut self) {
370        let commit_state = self
371            .pending_commits
372            .back_mut()
373            .unwrap_or_else(|| panic!("No pending commit."));
374        let mut committed_blocks = commit_state.commit.blocks.clone();
375        committed_blocks.sort_by_key(|b| b.round());
376        let blocks_map = self.blocks.read();
377        for committed_block in committed_blocks {
378            let committed_block_ref = committed_block.reference();
379            // Each block must have at least one ancestor.
380            // Block verification ensures the first ancestor is from the block's own authority.
381            // Also, block verification ensures each authority appears at most once among ancestors.
382            let mut origin_ancestor_ref = *blocks_map
383                .get(&committed_block_ref)
384                .unwrap()
385                .read()
386                .block
387                .ancestors()
388                .first()
389                .unwrap();
390            while origin_ancestor_ref.author == committed_block_ref.author {
391                let Some(origin_ancestor_block) = blocks_map.get(&origin_ancestor_ref) else {
392                    break;
393                };
394                origin_ancestor_block
395                    .write()
396                    .origin_descendants
397                    .push(committed_block_ref);
398                origin_ancestor_ref = *origin_ancestor_block
399                    .read()
400                    .block
401                    .ancestors()
402                    .first()
403                    .unwrap();
404            }
405        }
406    }

408    // Tries indirectly finalizing the buffered commits at the given index.
409    async fn try_indirect_finalize_first_commit(&mut self) {
410        // Ensure direct finalization has been attempted for the commit.
411        assert!(!self.pending_commits.is_empty());
412        assert!(self.pending_commits[0].pending_blocks.is_empty());
413
414        // Optional optimization: re-check pending transactions to see if they are rejected by a quorum now.
415        self.check_pending_transactions_in_first_commit();
416
417        // Check if remaining pending transactions can be finalized.
418        self.try_indirect_finalize_pending_transactions_in_first_commit()
419            .await;
420
421        // Check if remaining pending transactions can be indirectly rejected.
422        self.try_indirect_reject_pending_transactions_in_first_commit();
423    }
424
425    fn check_pending_transactions_in_first_commit(&mut self) {
426        let mut all_rejected_transactions: Vec<(BlockRef, Vec<TransactionIndex>)> = vec![];
427
428        // Collect all rejected transactions without modifying state
429        for (block_ref, pending_transactions) in &self.pending_commits[0].pending_transactions {
430            let reject_votes: BTreeMap<TransactionIndex, Stake> = self
431                .transaction_certifier
432                .get_reject_votes(block_ref)
433                .unwrap_or_else(|| panic!("No vote info found for {block_ref}. It is incorrectly gc'ed or failed to be recovered after crash."))
434                .into_iter()
435                .collect();
436            let mut rejected_transactions = vec![];
437            for &transaction_index in pending_transactions {
438                // Pending transactions should always have reject votes.
439                let reject_stake = reject_votes.get(&transaction_index).copied().unwrap();
440                if reject_stake < self.context.committee.quorum_threshold() {
441                    // The transaction cannot be rejected yet.
442                    continue;
443                }
444                // Otherwise, mark the transaction for rejection.
445                rejected_transactions.push(transaction_index);
446            }
447            if !rejected_transactions.is_empty() {
448                all_rejected_transactions.push((*block_ref, rejected_transactions));
449            }
450        }
451
452        // Move rejected transactions from pending_transactions.
453        for (block_ref, rejected_transactions) in all_rejected_transactions {
454            self.context
455                .metrics
456                .node_metrics
457                .finalizer_transaction_status
458                .with_label_values(&["direct_late_reject"])
459                .inc_by(rejected_transactions.len() as u64);
460            let curr_commit_state = &mut self.pending_commits[0];
461            curr_commit_state.remove_pending_transactions(&block_ref, &rejected_transactions);
462            curr_commit_state
463                .rejected_transactions
464                .entry(block_ref)
465                .or_default()
466                .extend(rejected_transactions);
467        }
468    }

470    async fn try_indirect_finalize_pending_transactions_in_first_commit(&mut self) {
471        let _scope = monitored_scope(
472            "CommitFinalizer::try_indirect_finalize_pending_transactions_in_first_commit",
473        );
474
475        let pending_blocks: Vec<(BlockRef, BTreeSet<TransactionIndex>)> = self.pending_commits[0]
476            .pending_transactions
477            .iter()
478            .map(|(k, v)| (*k, v.clone()))
479            .collect();
480
481        // Number of blocks to process in each task.
482        const BLOCKS_PER_INDIRECT_COMMIT_TASK: usize = 8;
483
484        // Process chunks in parallel.
485        let mut all_finalized_transactions = vec![];
486        let mut join_set = JoinSet::new();
487        // TODO(fastpath): investigate using a cost based batching,
488        // for example each block has cost num authorities + pending_transactions.len().
489        for chunk in pending_blocks.chunks(BLOCKS_PER_INDIRECT_COMMIT_TASK) {
490            let context = self.context.clone();
491            let blocks = self.blocks.clone();
492            let chunk: Vec<(BlockRef, BTreeSet<TransactionIndex>)> = chunk.to_vec();
493
494            join_set.spawn(tokio::task::spawn_blocking(move || {
495                let mut chunk_results = Vec::new();
496
497                for (block_ref, pending_transactions) in chunk {
498                    let finalized = Self::try_indirect_finalize_pending_transactions_in_block(
499                        &context,
500                        &blocks,
501                        block_ref,
502                        pending_transactions,
503                    );
504
505                    if !finalized.is_empty() {
506                        chunk_results.push((block_ref, finalized));
507                    }
508                }
509
510                chunk_results
511            }));
512        }
513
514        // Collect results from all chunks
515        while let Some(result) = join_set.join_next().await {
516            let e = match result {
517                Ok(blocking_result) => match blocking_result {
518                    Ok(chunk_results) => {
519                        all_finalized_transactions.extend(chunk_results);
520                        continue;
521                    }
522                    Err(e) => e,
523                },
524                Err(e) => e,
525            };
526            if e.is_panic() {
527                std::panic::resume_unwind(e.into_panic());
528            }
529            tracing::info!("Process likely shutting down: {:?}", e);
530            // Ok to return. No potential inconsistency in state.
531            return;
532        }
533
534        for (block_ref, finalized_transactions) in all_finalized_transactions {
535            self.context
536                .metrics
537                .node_metrics
538                .finalizer_transaction_status
539                .with_label_values(&["indirect_finalize"])
540                .inc_by(finalized_transactions.len() as u64);
541            // Remove finalized transactions from pending transactions.
542            self.pending_commits[0]
543                .remove_pending_transactions(&block_ref, &finalized_transactions);
544        }
545    }

547    fn try_indirect_reject_pending_transactions_in_first_commit(&mut self) {
548        let curr_leader_round = self.pending_commits[0].commit.leader.round;
549        let last_commit_leader_round = self.pending_commits.back().unwrap().commit.leader.round;
550        if curr_leader_round + INDIRECT_REJECT_DEPTH <= last_commit_leader_round {
551            let curr_commit_state = &mut self.pending_commits[0];
552            // This function is called after trying to indirectly finalize pending blocks.
553            // When last commit leader round is INDIRECT_REJECT_DEPTH rounds higher or more,
554            // all pending blocks should have been finalized.
555            assert!(curr_commit_state.pending_blocks.is_empty());
556            // This function is called after trying to indirectly finalize pending transactions.
557            // All remaining pending transactions, since they are not finalized, should now be
558            // indirectly rejected.
559            let pending_transactions = std::mem::take(&mut curr_commit_state.pending_transactions);
560            for (block_ref, pending_transactions) in pending_transactions {
561                self.context
562                    .metrics
563                    .node_metrics
564                    .finalizer_transaction_status
565                    .with_label_values(&["indirect_reject"])
566                    .inc_by(pending_transactions.len() as u64);
567                curr_commit_state
568                    .rejected_transactions
569                    .entry(block_ref)
570                    .or_default()
571                    .extend(pending_transactions);
572            }
573        }
574    }

576    // Returns the indices of the requested pending transactions that are indirectly finalized.
577    // This function is used for checking finalization of transactions, so it must traverse
578    // all blocks which can contribute to the requested transactions' finalizations.
579    fn try_indirect_finalize_pending_transactions_in_block(
580        context: &Arc<Context>,
581        blocks: &Arc<RwLock<BTreeMap<BlockRef, RwLock<BlockState>>>>,
582        pending_block_ref: BlockRef,
583        pending_transactions: BTreeSet<TransactionIndex>,
584    ) -> Vec<TransactionIndex> {
585        if pending_transactions.is_empty() {
586            return vec![];
587        }
588        let mut accept_votes: BTreeMap<TransactionIndex, StakeAggregator<QuorumThreshold>> =
589            pending_transactions
590                .into_iter()
591                .map(|transaction_index| (transaction_index, StakeAggregator::new()))
592                .collect();
593        let mut finalized_transactions = vec![];
594        let blocks_map = blocks.read();
595        // Use BTreeSet to ensure always visit blocks in the earliest round.
596        let mut to_visit_blocks = blocks_map
597            .get(&pending_block_ref)
598            .unwrap()
599            .read()
600            .children
601            .clone();
602        // Blocks that have been visited.
603        let mut visited = BTreeSet::new();
604        // Blocks where votes and origin descendants should be ignored for processing.
605        let mut ignored = BTreeSet::new();
606        // Traverse children blocks breadth-first and accumulate accept votes for pending transactions.
607        while let Some(curr_block_ref) = to_visit_blocks.pop_first() {
608            if !visited.insert(curr_block_ref) {
609                continue;
610            }
611            let curr_block_state = blocks_map.get(&curr_block_ref).unwrap_or_else(|| panic!("Block {curr_block_ref} is either incorrectly gc'ed or failed to be recovered after crash.")).read();
612            // Ignore info from the block if its direct ancestor has been processed.
613            if ignored.insert(curr_block_ref) {
614                // Skip collecting votes from origin descendants of current block.
615                // Votes from origin descendants of current block do not count for this transactions.
616                // Consider this case: block B is an origin descendant of block A (from the same authority),
617                // and both blocks A and B link to another block C.
618                // Only B's implicit and explicit transaction votes on C are considered.
619                // None of A's implicit or explicit transaction votes on C should be considered.
620                ignored.extend(curr_block_state.origin_descendants.iter());
621                // Get reject votes from current block to the pending block.
622                let curr_block_reject_votes = curr_block_state
623                    .reject_votes
624                    .get(&pending_block_ref)
625                    .cloned()
626                    .unwrap_or_default();
627                // Because of lifetime, first collect finalized transactions, and then remove them from accept_votes.
628                let mut newly_finalized = vec![];
629                for (index, stake) in &mut accept_votes {
630                    // Skip if the transaction has been rejected by the current block.
631                    if curr_block_reject_votes.contains(index) {
632                        continue;
633                    }
634                    // Skip if the total stake has not reached quorum.
635                    if !stake.add(curr_block_ref.author, &context.committee) {
636                        continue;
637                    }
638                    newly_finalized.push(*index);
639                    finalized_transactions.push(*index);
640                }
641                // There is no need to aggregate additional votes for already finalized transactions.
642                for index in newly_finalized {
643                    accept_votes.remove(&index);
644                }
645                // End traversing if all blocks and requested transactions have reached quorum.
646                if accept_votes.is_empty() {
647                    break;
648                }
649            }
650            // Add additional children blocks to visit.
651            to_visit_blocks.extend(
652                curr_block_state
653                    .children
654                    .iter()
655                    .filter(|b| !visited.contains(*b)),
656            );
657        }
658        finalized_transactions
659    }
660
661    fn pop_finalized_commits(&mut self) -> Vec<CommittedSubDag> {
662        let mut finalized_commits = vec![];
663
664        while let Some(commit_state) = self.pending_commits.front() {
665            if !commit_state.pending_blocks.is_empty()
666                || !commit_state.pending_transactions.is_empty()
667            {
668                // The commit is not finalized yet.
669                break;
670            }
671
672            // Pop the finalized commit and set its rejected transactions.
673            let commit_state = self.pending_commits.pop_front().unwrap();
674            let mut commit = commit_state.commit;
675            for (block_ref, rejected_transactions) in commit_state.rejected_transactions {
676                commit
677                    .rejected_transactions_by_block
678                    .insert(block_ref, rejected_transactions.into_iter().collect());
679            }
680
681            // Clean up committed blocks.
682            let mut blocks_map = self.blocks.write();
683            for block in commit.blocks.iter() {
684                blocks_map.remove(&block.reference());
685            }
686
687            let round_delay = if let Some(last_commit_state) = self.pending_commits.back() {
688                last_commit_state.commit.leader.round - commit.leader.round
689            } else {
690                0
691            };
692            self.context
693                .metrics
694                .node_metrics
695                .finalizer_round_delay
696                .observe(round_delay as f64);
697
698            finalized_commits.push(commit);
699        }
700
701        finalized_commits
702    }
703
704    #[cfg(test)]
705    fn is_empty(&self) -> bool {
706        self.pending_commits.is_empty() && self.blocks.read().is_empty()
707    }

710struct CommitState {
711    commit: CommittedSubDag,
712    // Blocks pending finalization, mapped to the number of transactions in the block.
713    // This field is populated by all blocks in the commit, before direct finalization.
714    // After direct finalization, this field becomes empty.
715    pending_blocks: BTreeMap<BlockRef, usize>,
716    // Transactions pending indirect finalization.
717    // This field is populated after direct finalization, if pending transactions exist.
718    // Values in this field are removed as transactions are indirectly finalized or directly rejected.
719    // When both pending_blocks and pending_transactions are empty, the commit is finalized.
720    pending_transactions: BTreeMap<BlockRef, BTreeSet<TransactionIndex>>,
721    // Transactions rejected by a quorum or indirectly, per block.
722    rejected_transactions: BTreeMap<BlockRef, BTreeSet<TransactionIndex>>,
723}
724
725impl CommitState {
726    fn new(commit: CommittedSubDag) -> Self {
727        let pending_blocks: BTreeMap<_, _> = commit
728            .blocks
729            .iter()
730            .map(|b| (b.reference(), b.transactions().len()))
731            .collect();
732        assert!(!pending_blocks.is_empty());
733        Self {
734            commit,
735            pending_blocks,
736            pending_transactions: BTreeMap::new(),
737            rejected_transactions: BTreeMap::new(),
738        }
739    }
740
741    fn remove_pending_transactions(
742        &mut self,
743        block_ref: &BlockRef,
744        transactions: &[TransactionIndex],
745    ) {
746        let Some(block_pending_txns) = self.pending_transactions.get_mut(block_ref) else {
747            return;
748        };
749        for t in transactions {
750            block_pending_txns.remove(t);
751        }
752        if block_pending_txns.is_empty() {
753            self.pending_transactions.remove(block_ref);
754        }
755    }
756}

758struct BlockState {
759    // Content of the block.
760    block: VerifiedBlock,
761    // Blocks which has an explicit ancestor linking to this block.
762    children: BTreeSet<BlockRef>,
763    // Reject votes casted by this block, and by linked ancestors from the same authority.
764    reject_votes: BTreeMap<BlockRef, BTreeSet<TransactionIndex>>,
765    // Other committed blocks that are origin descendants of this block.
766    origin_descendants: Vec<BlockRef>,
767}
768
769impl BlockState {
770    fn new(block: VerifiedBlock) -> Self {
771        let reject_votes: BTreeMap<_, _> = block
772            .transaction_votes()
773            .iter()
774            .map(|v| (v.block_ref, v.rejects.clone().into_iter().collect()))
775            .collect();
776        // With at most 4 pending commits and assume 2 origin descendants per commit,
777        // there will be at most 8 origin descendants.
778        let origin_descendants = Vec::with_capacity(8);
779        Self {
780            block,
781            children: BTreeSet::new(),
782            reject_votes,
783            origin_descendants,
784        }
785    }

1212            leaders: &[VerifiedBlock],
1213            all_blocks: &[Vec<VerifiedBlock>],
1214            index: usize,
1215            local: bool,
1216        ) -> Vec<CommittedSubDag> {
1217            let leader = leaders[index].clone();
1218            // Add blocks related to the commit to DagState and TransactionCertifier.
1219            if local {
1220                for round_blocks in all_blocks.iter().take(leader.round() as usize + 2) {
1221                    fixture.add_blocks(round_blocks.clone());
1222                }
1223            } else {
1224                for round_blocks in all_blocks.iter().take(leader.round() as usize) {
1225                    fixture.add_blocks(round_blocks.clone());
1226                }
1227            };
1228            // Generate remote commit from leader.
1229            let mut committed_sub_dags = fixture.linearizer.handle_commit(vec![leader]);
1230            assert_eq!(committed_sub_dags.len(), 1);
1231            let mut remote_commit = committed_sub_dags.pop().unwrap();
1232            remote_commit.decided_with_local_blocks = local;
1233            // Process the remote commit.
1234            fixture
1235                .commit_finalizer
1236                .process_commit(remote_commit.clone())
1237                .await
1238        }
1239
1240        // Add commit 1-3 as remote commits. There should be no finalized commits.
1241        for i in 0..3 {
1242            let finalized_commits =
1243                add_blocks_and_process_commit(&mut fixture, &leaders, &all_blocks, i, false).await;
1244            assert!(finalized_commits.is_empty());
1245        }
1246
1247        // Buffer round 4 commit as a remote commit. This should finalize the 1st commit at round 1.
1248        let finalized_commits =
1249            add_blocks_and_process_commit(&mut fixture, &leaders, &all_blocks, 3, false).await;
1250        assert_eq!(finalized_commits.len(), 1);
1251        assert_eq!(finalized_commits[0].commit_ref.index, 1);
1252        assert_eq!(finalized_commits[0].leader.round, 1);
1253
1254        // Buffer round 6 (5th) commit as local commit. This should help finalize the commits at round 2 and 3.
1255        let finalized_commits =
1256            add_blocks_and_process_commit(&mut fixture, &leaders, &all_blocks, 4, true).await;
1257        assert_eq!(finalized_commits.len(), 2);
1258        assert_eq!(finalized_commits[0].commit_ref.index, 2);
1259        assert_eq!(finalized_commits[0].leader.round, 2);
1260        assert_eq!(finalized_commits[1].commit_ref.index, 3);
1261        assert_eq!(finalized_commits[1].leader.round, 3);
1262
1263        // Buffer round 7 (6th) commit as local commit. This should help finalize the commits at round 4, 6 and 7 (itself).
1264        let finalized_commits =
1265            add_blocks_and_process_commit(&mut fixture, &leaders, &all_blocks, 5, true).await;
1266        assert_eq!(finalized_commits.len(), 3);
1267        assert_eq!(finalized_commits[0].commit_ref.index, 4);
1268        assert_eq!(finalized_commits[0].leader.round, 4);
1269        assert_eq!(finalized_commits[1].commit_ref.index, 5);
1270        assert_eq!(finalized_commits[1].leader.round, 6);
1271        assert_eq!(finalized_commits[2].commit_ref.index, 6);
1272        assert_eq!(finalized_commits[2].leader.round, 7);
1273
1274        // CommitFinalizer should be empty.
1275        assert!(fixture.commit_finalizer.is_empty());
1276    }
