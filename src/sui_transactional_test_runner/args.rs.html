313            }
314            Some(("consensus-commit-prologue", matches)) => SuiSubcommand::ConsensusCommitPrologue(
315                ConsensusCommitPrologueCommand::from_arg_matches(matches)?,
316            ),
317            Some(("programmable", matches)) => SuiSubcommand::ProgrammableTransaction(
318                ProgrammableTransactionCommand::from_arg_matches(matches)?,
319            ),
320            Some(("upgrade", matches)) => {
321                SuiSubcommand::UpgradePackage(UpgradePackageCommand::from_arg_matches(matches)?)
322            }
323            Some(("stage-package", matches)) => {
324                SuiSubcommand::StagePackage(StagePackageCommand::from_arg_matches(matches)?)
325            }
326            Some(("set-address", matches)) => {
327                SuiSubcommand::SetAddress(SetAddressCommand::from_arg_matches(matches)?)
328            }
329            Some(("create-checkpoint", matches)) => {
330                SuiSubcommand::CreateCheckpoint(CreateCheckpointCommand::from_arg_matches(matches)?)
331            }
332            Some(("advance-epoch", matches)) => {
333                SuiSubcommand::AdvanceEpoch(AdvanceEpochCommand::from_arg_matches(matches)?)
334            }
335            Some(("advance-clock", matches)) => {
336                SuiSubcommand::AdvanceClock(AdvanceClockCommand::from_arg_matches(matches)?)
337            }
338            Some(("set-random-state", matches)) => {
339                SuiSubcommand::SetRandomState(SetRandomStateCommand::from_arg_matches(matches)?)
340            }
341            Some(("authenticator-state-update", matches)) => {
342                SuiSubcommand::AuthenticatorStateUpdate(
343                    AuthenticatorStateUpdateCommand::from_arg_matches(matches)?,
344                )
345            }
346            Some(("view-checkpoint", _)) => SuiSubcommand::ViewCheckpoint,
347            Some(("run-graphql", matches)) => {
348                SuiSubcommand::RunGraphql(RunGraphqlCommand::from_arg_matches(matches)?)
349            }
350            Some(("run-jsonrpc", matches)) => {
351                SuiSubcommand::RunJsonRpc(RunJsonRpcCommand::from_arg_matches(matches)?)
352            }
353            Some(("bench", matches)) => SuiSubcommand::Bench(
354                RunCommand::from_arg_matches(matches)?,
355                ExtraRunArgs::from_arg_matches(matches)?,
356            ),
357            _ => {
358                return Err(clap::Error::raw(
359                    clap::error::ErrorKind::InvalidSubcommand,
360                    "Invalid submcommand",
361                ));
362            }
363        })
364    }
365
366    fn update_from_arg_matches(&mut self, matches: &clap::ArgMatches) -> Result<(), clap::Error> {
367        *self = Self::from_arg_matches(matches)?;
368        Ok(())
369    }
370}

418#[derive(Clone)]
419pub enum SuiValue {
420    MoveValue(MoveValue),
421    Object(FakeID, Option<SequenceNumber>),
422    ObjVec(Vec<(FakeID, Option<SequenceNumber>)>),
423    Digest(String),

428
429impl SuiExtraValueArgs {
430    fn parse_object_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
431        parser: &mut MoveCLParser<'a, ValueToken, I>,
432    ) -> anyhow::Result<Self> {
433        let (fake_id, version) = Self::parse_receiving_or_object_value(parser, "object")?;
434        Ok(SuiExtraValueArgs::Object(fake_id, version))
435    }
436
437    fn parse_receiving_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
438        parser: &mut MoveCLParser<'a, ValueToken, I>,
439    ) -> anyhow::Result<Self> {
440        let (fake_id, version) = Self::parse_receiving_or_object_value(parser, "receiving")?;
441        Ok(SuiExtraValueArgs::Receiving(fake_id, version))
442    }
443
444    fn parse_read_shared_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
445        parser: &mut MoveCLParser<'a, ValueToken, I>,
446    ) -> anyhow::Result<Self> {
447        let (fake_id, version) = Self::parse_receiving_or_object_value(parser, "immshared")?;
448        Ok(SuiExtraValueArgs::ImmShared(fake_id, version))
449    }
450
451    fn parse_non_exlucsive_write_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
452        parser: &mut MoveCLParser<'a, ValueToken, I>,
453    ) -> anyhow::Result<Self> {
454        let (fake_id, version) = Self::parse_receiving_or_object_value(parser, "nonexclusive")?;
455        Ok(SuiExtraValueArgs::NonExclusiveWrite(fake_id, version))
456    }
457
458    fn parse_digest_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
459        parser: &mut MoveCLParser<'a, ValueToken, I>,
460    ) -> anyhow::Result<Self> {
461        let contents = parser.advance(ValueToken::Ident)?;
462        ensure!(contents == "digest");
463        parser.advance(ValueToken::LParen)?;
464        let package = parser.advance(ValueToken::Ident)?;
465        parser.advance(ValueToken::RParen)?;
466        Ok(SuiExtraValueArgs::Digest(package.to_owned()))
467    }
468
469    fn parse_receiving_or_object_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
470        parser: &mut MoveCLParser<'a, ValueToken, I>,
471        ident_name: &str,
472    ) -> anyhow::Result<(FakeID, Option<SequenceNumber>)> {
473        let contents = parser.advance(ValueToken::Ident)?;
474        ensure!(contents == ident_name);
475        parser.advance(ValueToken::LParen)?;
476        let i_str = parser.advance(ValueToken::Number)?;
477        let (i, _) = parse_u256(i_str)?;
478        let fake_id = if let Some(ValueToken::Comma) = parser.peek_tok() {
479            parser.advance(ValueToken::Comma)?;
480            let j_str = parser.advance(ValueToken::Number)?;
481            let (j, _) = parse_u64(j_str)?;
482            if i > U256::from(u64::MAX) {
483                bail!("Object ID too large")
484            }
485            FakeID::Enumerated(i.unchecked_as_u64(), j)
486        } else {
487            let mut u256_bytes = i.to_le_bytes().to_vec();
488            u256_bytes.reverse();
489            let address: SuiAddress = SuiAddress::from_bytes(&u256_bytes).unwrap();
490            FakeID::Known(address.into())
491        };
492        parser.advance(ValueToken::RParen)?;
493        let version = if let Some(ValueToken::AtSign) = parser.peek_tok() {
494            parser.advance(ValueToken::AtSign)?;
495            let v_str = parser.advance(ValueToken::Number)?;
496            let (v, _) = parse_u64(v_str)?;
497            Some(SequenceNumber::from_u64(v))
498        } else {
499            None
500        };
501        Ok((fake_id, version))
502    }
503}

505impl SuiValue {
506    fn assert_move_value(self) -> MoveValue {
507        match self {
508            SuiValue::MoveValue(v) => v,
509            SuiValue::Object(_, _) => panic!("unexpected nested Sui object in args"),
510            SuiValue::ObjVec(_) => panic!("unexpected nested Sui object vector in args"),
511            SuiValue::Digest(_) => panic!("unexpected nested Sui package digest in args"),
512            SuiValue::Receiving(_, _) => panic!("unexpected nested Sui receiving object in args"),
513            SuiValue::ImmShared(_, _) => panic!("unexpected nested Sui shared object in args"),
514            SuiValue::NonExclusiveWrite(_, _) => {
515                panic!("unexpected nested Sui non-exclusive write object in args")
516            }
517        }
518    }
519
520    fn assert_object(self) -> (FakeID, Option<SequenceNumber>) {
521        match self {
522            SuiValue::MoveValue(_) => panic!("unexpected nested non-object value in args"),
523            SuiValue::Object(id, version) => (id, version),
524            SuiValue::ObjVec(_) => panic!("unexpected nested Sui object vector in args"),
525            SuiValue::Digest(_) => panic!("unexpected nested Sui package digest in args"),
526            SuiValue::Receiving(_, _) => panic!("unexpected nested Sui receiving object in args"),
527            SuiValue::ImmShared(_, _) => panic!("unexpected nested Sui shared object in args"),
528            SuiValue::NonExclusiveWrite(_, _) => {
529                panic!("unexpected nested Sui non-exclusive write object in args")
530            }
531        }
532    }
533
534    fn resolve_object(
535        fake_id: FakeID,
536        version: Option<SequenceNumber>,
537        test_adapter: &SuiTestAdapter,
538    ) -> anyhow::Result<Object> {
539        let id = match test_adapter.fake_to_real_object_id(fake_id) {
540            Some(id) => id,
541            None => bail!("INVALID TEST. Unknown object, object({})", fake_id),
542        };
543        let obj_res = if let Some(v) = version {
544            sui_types::storage::ObjectStore::get_object_by_key(&*test_adapter.executor, &id, v)
545        } else {
546            sui_types::storage::ObjectStore::get_object(&*test_adapter.executor, &id)
547        };
548        let obj = match obj_res {
549            Some(obj) => obj,
550            None => bail!("INVALID TEST. Could not load object argument {}", id),
551        };
552        Ok(obj)
553    }
554
555    fn receiving_arg(
556        fake_id: FakeID,
557        version: Option<SequenceNumber>,
558        test_adapter: &SuiTestAdapter,
559    ) -> anyhow::Result<ObjectArg> {
560        let obj = Self::resolve_object(fake_id, version, test_adapter)?;
561        Ok(ObjectArg::Receiving(obj.compute_object_reference()))
562    }
563
564    fn read_shared_arg(
565        fake_id: FakeID,
566        version: Option<SequenceNumber>,
567        test_adapter: &SuiTestAdapter,
568    ) -> anyhow::Result<ObjectArg> {
569        Self::shared_arg_impl(
570            fake_id,
571            version,
572            test_adapter,
573            SharedObjectMutability::Immutable,
574        )
575    }
576
577    fn non_exclusive_write_arg(
578        fake_id: FakeID,
579        version: Option<SequenceNumber>,
580        test_adapter: &SuiTestAdapter,
581    ) -> anyhow::Result<ObjectArg> {
582        Self::shared_arg_impl(
583            fake_id,
584            version,
585            test_adapter,
586            SharedObjectMutability::NonExclusiveWrite,
587        )
588    }
589
590    fn shared_arg_impl(
591        fake_id: FakeID,
592        version: Option<SequenceNumber>,
593        test_adapter: &SuiTestAdapter,
594        mutability: SharedObjectMutability,
595    ) -> anyhow::Result<ObjectArg> {
596        let obj = Self::resolve_object(fake_id, version, test_adapter)?;
597        let id = obj.id();
598        if let Owner::Shared {
599            initial_shared_version,
600        } = obj.owner
601        {
602            Ok(ObjectArg::SharedObject {
603                id,
604                initial_shared_version,
605                mutability,
606            })
607        } else {
608            bail!("{fake_id} is not a shared object.")
609        }
610    }
611
612    fn object_arg(
613        fake_id: FakeID,
614        version: Option<SequenceNumber>,
615        test_adapter: &SuiTestAdapter,
616    ) -> anyhow::Result<ObjectArg> {
617        let obj = Self::resolve_object(fake_id, version, test_adapter)?;
618        let id = obj.id();
619        match obj.owner {
620            Owner::Shared {
621                initial_shared_version,
622            }
623            | Owner::ConsensusAddressOwner {
624                start_version: initial_shared_version,
625                ..
626            } => Ok(ObjectArg::SharedObject {
627                id,
628                initial_shared_version,
629                mutability: SharedObjectMutability::Mutable,
630            }),
631            Owner::AddressOwner(_) | Owner::ObjectOwner(_) | Owner::Immutable => {
632                let obj_ref = obj.compute_object_reference();
633                Ok(ObjectArg::ImmOrOwnedObject(obj_ref))
634            }
635        }
636    }
637
638    pub(crate) fn into_call_arg(self, test_adapter: &SuiTestAdapter) -> anyhow::Result<CallArg> {
639        Ok(match self {
640            SuiValue::Object(fake_id, version) => {
641                CallArg::Object(Self::object_arg(fake_id, version, test_adapter)?)
642            }
643            SuiValue::MoveValue(v) => CallArg::Pure(v.simple_serialize().unwrap()),
644            SuiValue::Receiving(fake_id, version) => {
645                CallArg::Object(Self::receiving_arg(fake_id, version, test_adapter)?)
646            }
647            SuiValue::ImmShared(fake_id, version) => {
648                CallArg::Object(Self::read_shared_arg(fake_id, version, test_adapter)?)
649            }
650            SuiValue::NonExclusiveWrite(fake_id, version) => CallArg::Object(
651                Self::non_exclusive_write_arg(fake_id, version, test_adapter)?,
652            ),
653            SuiValue::ObjVec(_) => bail!("obj vec is not supported as an input"),
654            SuiValue::Digest(pkg) => {
655                let pkg = Symbol::from(pkg);
656                let Some(staged) = test_adapter.staged_modules.get(&pkg) else {
657                    bail!("Unbound staged package '{pkg}'")
658                };
659                CallArg::Pure(bcs::to_bytes(&staged.digest).unwrap())
660            }
661        })
662    }
663
664    pub(crate) fn into_argument(
665        self,
666        builder: &mut ProgrammableTransactionBuilder,
667        test_adapter: &SuiTestAdapter,
668    ) -> anyhow::Result<Argument> {
669        match self {
670            SuiValue::ObjVec(vec) => builder.make_obj_vec(
671                vec.iter()
672                    .map(|(fake_id, version)| Self::object_arg(*fake_id, *version, test_adapter))
673                    .collect::<Result<Vec<ObjectArg>, _>>()?,
674            ),
675            value => {
676                let call_arg = value.into_call_arg(test_adapter)?;
677                builder.input(call_arg)
678            }
679        }
680    }
681}

683impl ParsableValue for SuiExtraValueArgs {
684    type ConcreteValue = SuiValue;
685
686    fn parse_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
687        parser: &mut MoveCLParser<'a, ValueToken, I>,
688    ) -> Option<anyhow::Result<Self>> {
689        match parser.peek()? {
690            (ValueToken::Ident, "object") => Some(Self::parse_object_value(parser)),
691            (ValueToken::Ident, "digest") => Some(Self::parse_digest_value(parser)),
692            (ValueToken::Ident, "receiving") => Some(Self::parse_receiving_value(parser)),
693            (ValueToken::Ident, "immshared") => Some(Self::parse_read_shared_value(parser)),
694            (ValueToken::Ident, "nonexclusive") => {
695                Some(Self::parse_non_exlucsive_write_value(parser))
696            }
697            _ => None,
698        }
699    }
700
701    fn move_value_into_concrete(v: MoveValue) -> anyhow::Result<Self::ConcreteValue> {
702        Ok(SuiValue::MoveValue(v))
703    }
704
705    fn concrete_vector(elems: Vec<Self::ConcreteValue>) -> anyhow::Result<Self::ConcreteValue> {
706        if !elems.is_empty() && matches!(elems[0], SuiValue::Object(_, _)) {
707            Ok(SuiValue::ObjVec(
708                elems.into_iter().map(SuiValue::assert_object).collect(),
709            ))
710        } else {
711            Ok(SuiValue::MoveValue(MoveValue::Vector(
712                elems.into_iter().map(SuiValue::assert_move_value).collect(),
713            )))
714        }
715    }
716
717    fn concrete_struct(values: Vec<Self::ConcreteValue>) -> anyhow::Result<Self::ConcreteValue> {
718        Ok(SuiValue::MoveValue(MoveValue::Struct(MoveStruct(
719            values.into_iter().map(|v| v.assert_move_value()).collect(),
720        ))))
721    }
722
723    fn into_concrete_value(
724        self,
725        _mapping: &impl Fn(&str) -> Option<move_core_types::account_address::AccountAddress>,
726    ) -> anyhow::Result<Self::ConcreteValue> {
727        match self {
728            SuiExtraValueArgs::Object(id, version) => Ok(SuiValue::Object(id, version)),
729            SuiExtraValueArgs::Digest(pkg) => Ok(SuiValue::Digest(pkg)),
730            SuiExtraValueArgs::Receiving(id, version) => Ok(SuiValue::Receiving(id, version)),
731            SuiExtraValueArgs::ImmShared(id, version) => Ok(SuiValue::ImmShared(id, version)),
732            SuiExtraValueArgs::NonExclusiveWrite(id, version) => {
733                Ok(SuiValue::NonExclusiveWrite(id, version))
734            }
735        }
736    }
737}

739fn parse_fake_id(s: &str) -> anyhow::Result<FakeID> {
740    Ok(if let Some((s1, s2)) = s.split_once(',') {
741        let (i, _) = parse_u64(s1)?;
742        let (j, _) = parse_u64(s2)?;
743        FakeID::Enumerated(i, j)
744    } else {
745        let (i, _) = parse_u256(s)?;
746        let mut u256_bytes = i.to_le_bytes().to_vec();
747        u256_bytes.reverse();
748        let address: SuiAddress = SuiAddress::from_bytes(&u256_bytes).unwrap();
749        FakeID::Known(address.into())
750    })
751}

753fn parse_policy(x: &str) -> anyhow::Result<u8> {
754    Ok(match x {
755        "compatible" => UpgradePolicy::COMPATIBLE,
756        "additive" => UpgradePolicy::ADDITIVE,
757        "dep_only" => UpgradePolicy::DEP_ONLY,
758        _ => bail!(
759            "Invalid upgrade policy {x}. Policy must be one of 'compatible', 'additive', or 'dep_only'"
760        ),
761    })
